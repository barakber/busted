5use tokio::sync::mpsc;

9const SOCKET_PATH: &str = "/tmp/busted.sock";
10
11/// Connect to the agent's Unix socket and forward NDJSON events.
12/// Reconnects automatically on disconnection.
13pub async fn unix_socket_bridge(tx: mpsc::UnboundedSender<AppEvent>) {
14    loop {
15        match UnixStream::connect(SOCKET_PATH).await {
16            Ok(stream) => {
17                log::info!("Connected to agent at {SOCKET_PATH}");
18                let reader = BufReader::new(stream);
19                let mut lines = reader.lines();
20                while let Ok(Some(line)) = lines.next_line().await {
21                    if line.is_empty() {
22                        continue;
23                    }
24                    match serde_json::from_str::<BustedEvent>(&line) {
25                        Ok(event) => {
26                            if tx.send(AppEvent::Busted(Box::new(event))).is_err() {
27                                return;
28                            }
29                        }
30                        Err(e) => {
31                            warn!("Failed to parse event: {e}");
32                        }
33                    }
34                }
35                log::info!("Agent connection closed, reconnecting...");
36            }
37            Err(_) => {
38                debug!("Agent not available at {SOCKET_PATH}, retrying...");
39            }
40        }
41        tokio::time::sleep(std::time::Duration::from_secs(2)).await;
42    }
43}
