13pub fn draw(f: &mut Frame, area: ratatui::layout::Rect, app: &mut App) {
14    let filtered = app.filtered_events();
15
16    let header = Row::new(vec![
17        Cell::from("Time"),
18        Cell::from("Action"),
19        Cell::from("Process"),
20        Cell::from("PID"),
21        Cell::from("Provider"),
22        Cell::from("Destination"),
23        Cell::from("Bytes"),
24        Cell::from("Pol"),
25    ])
26    .style(
27        Style::default()
28            .fg(style::BRIGHT_TEXT)
29            .add_modifier(Modifier::BOLD),
30    )
31    .height(1);
32
33    let rows: Vec<Row> = filtered.iter().map(|ev| event_row(ev)).collect();
34
35    let widths = [
36        Constraint::Length(12), // Time (fixed)
37        Constraint::Length(14), // Action: arrow + type (fixed)
38        Constraint::Fill(1),    // Process (flexible)
39        Constraint::Length(6),  // PID (fixed)
40        Constraint::Fill(1),    // Provider (flexible)
41        Constraint::Fill(2),    // Destination (flexible, gets 2x share)
42        Constraint::Length(7),  // Bytes (fixed)
43        Constraint::Length(5),  // Policy (fixed)
44    ];
45
46    let table = Table::new(rows, widths)
47        .header(header)
48        .block(Block::default().borders(Borders::NONE))
49        .row_highlight_style(style::selected_style());
50
51    f.render_stateful_widget(table, area, &mut app.table_state);
52}

54fn event_row(ev: &BustedEvent) -> Row<'static> {
55    let action_color = style::action_color(&ev.action);
56    let arrow = action_arrow(&ev.action);
57    let action_type = ev.action_type();
58
59    let time = ev.timestamp.clone();
60    let time_short = if time.len() > 12 {
61        time[..12].to_string()
62    } else {
63        time
64    };
65
66    let provider = ev.provider().unwrap_or("-").to_string();
67    let dest = ev.sni().unwrap_or("-").to_string();
68    let bytes = ev.bytes();
69    let bytes_str = if bytes > 0 {
70        format_bytes(bytes)
71    } else {
72        "-".to_string()
73    };
74    let policy = ev.policy.as_deref().unwrap_or("-").to_string();
75    let policy_color = style::policy_color(ev.policy.as_deref());
76
77    let action_cell = Cell::from(Line::from(vec![
78        Span::styled(format!("{arrow} "), Style::default().fg(action_color)),
79        Span::styled(action_type.to_string(), Style::default().fg(action_color)),
80    ]));
81
82    let row = Row::new(vec![
83        Cell::from(Span::styled(time_short, style::dim_style())),
84        action_cell,
85        Cell::from(Span::styled(ev.process.name.clone(), style::normal_style())),
86        Cell::from(Span::styled(ev.process.pid.to_string(), style::dim_style())),
87        Cell::from(Span::styled(provider, style::normal_style())),
88        Cell::from(Span::styled(dest, style::dim_style())),
89        Cell::from(Span::styled(bytes_str, style::normal_style())),
90        Cell::from(Span::styled(policy, Style::default().fg(policy_color))),
91    ]);
92
93    if ev.pii_detected() {
94        row.style(style::pii_row_style())
95    } else {
96        row
97    }
98}
