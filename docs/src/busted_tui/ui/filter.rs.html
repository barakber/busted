1use ratatui::{
2    layout::Rect,
3    style::{Modifier, Style},
4    text::{Line, Span},
5    Frame,
6};
7
8use crate::app::{App, InputMode};
9use crate::ui::style as s;
10
11pub fn draw(f: &mut Frame, area: Rect, app: &App) {
12    let mut spans = Vec::new();
13
14    // Search field
15    match app.input_mode {
16        InputMode::Search => {
17            spans.push(Span::styled(
18                " /",
19                Style::default()
20                    .fg(s::ACCENT_COLOR)
21                    .add_modifier(Modifier::BOLD),
22            ));
23            spans.push(Span::styled(
24                &app.search_query,
25                Style::default().fg(s::BRIGHT_TEXT),
26            ));
27            spans.push(Span::styled(
28                "\u{2588}",
29                Style::default().fg(s::ACCENT_COLOR),
30            )); // cursor block
31        }
32        InputMode::Normal => {
33            if app.search_query.is_empty() {
34                spans.push(Span::styled(" /search\u{2026}", s::dim_style()));
35            } else {
36                spans.push(Span::styled(" /", s::dim_style()));
37                spans.push(Span::styled(
38                    &app.search_query,
39                    Style::default().fg(s::ACCENT_COLOR),
40                ));
41            }
42        }
43    }
44
45    spans.push(Span::styled("  ", s::dim_style()));
46
47    // Provider filter
48    let prov = app.provider_filter.as_deref().unwrap_or("All");
49    spans.push(Span::styled("Provider:", s::dim_style()));
50    spans.push(Span::styled(
51        prov,
52        if app.provider_filter.is_some() {
53            Style::default().fg(s::ACCENT_COLOR)
54        } else {
55            s::dim_style()
56        },
57    ));
58
59    spans.push(Span::styled("  ", s::dim_style()));
60
61    // Type filter
62    let typ = app.type_filter.as_deref().unwrap_or("All");
63    spans.push(Span::styled("Type:", s::dim_style()));
64    spans.push(Span::styled(
65        typ,
66        if app.type_filter.is_some() {
67            Style::default().fg(s::ACCENT_COLOR)
68        } else {
69            s::dim_style()
70        },
71    ));
72
73    if app.has_active_filters() {
74        spans.push(Span::styled(
75            "  [C:clear]",
76            Style::default().fg(s::POLICY_AUDIT_COLOR),
77        ));
78    }
79
80    let line = Line::from(spans);
81    f.render_widget(line, area);
82}
