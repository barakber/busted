1use ratatui::{
2    layout::Constraint,
3    style::{Modifier, Style},
4    text::Span,
5    widgets::{Block, Borders, Cell, Row, Table},
6    Frame,
7};
8
9use crate::app::{format_bytes, App};
10use crate::ui::style as s;
11
12pub fn draw(f: &mut Frame, area: ratatui::layout::Rect, app: &mut App) {
13    let stats = app.provider_stats();
14
15    let header = Row::new(vec![
16        Cell::from("Provider"),
17        Cell::from("Events"),
18        Cell::from("Bytes"),
19        Cell::from("PII"),
20        Cell::from("Processes"),
21    ])
22    .style(
23        Style::default()
24            .fg(s::BRIGHT_TEXT)
25            .add_modifier(Modifier::BOLD),
26    )
27    .height(1);
28
29    let rows: Vec<Row> = stats
30        .iter()
31        .map(|stat| {
32            let pii_style = if stat.pii_count > 0 {
33                Style::default().fg(s::PII_COLOR)
34            } else {
35                s::dim_style()
36            };
37            Row::new(vec![
38                Cell::from(Span::styled(stat.name.clone(), s::normal_style())),
39                Cell::from(Span::styled(
40                    stat.event_count.to_string(),
41                    s::normal_style(),
42                )),
43                Cell::from(Span::styled(format_bytes(stat.bytes), s::normal_style())),
44                Cell::from(Span::styled(stat.pii_count.to_string(), pii_style)),
45                Cell::from(Span::styled(stat.processes.join(", "), s::dim_style())),
46            ])
47        })
48        .collect();
49
50    let widths = [
51        Constraint::Fill(1),    // Provider (flexible)
52        Constraint::Length(8),  // Events (fixed)
53        Constraint::Length(10), // Bytes (fixed)
54        Constraint::Length(5),  // PII (fixed)
55        Constraint::Fill(2),    // Processes (flexible, gets 2x share)
56    ];
57
58    let table = Table::new(rows, widths)
59        .header(header)
60        .block(Block::default().borders(Borders::NONE))
61        .row_highlight_style(s::selected_style());
62
63    f.render_stateful_widget(table, area, &mut app.provider_table_state);
64}
