43pub fn draw(f: &mut Frame, area: Rect) {
44    let popup_area = centered_rect(50, 70, area);
45    f.render_widget(Clear, popup_area);
46
47    let mut lines = vec![
48        Line::from(""),
49        Line::from(Span::styled(
50            "  Key Bindings",
51            Style::default()
52                .fg(s::ACCENT_COLOR)
53                .add_modifier(Modifier::BOLD),
54        )),
55        Line::from(""),
56    ];
57
58    for (key, desc) in BINDINGS {
59        if key.is_empty() {
60            lines.push(Line::from(""));
61            continue;
62        }
63        if desc.is_empty() {
64            lines.push(Line::from(Span::styled(
65                format!("  {key}"),
66                Style::default()
67                    .fg(s::BRIGHT_TEXT)
68                    .add_modifier(Modifier::BOLD),
69            )));
70            continue;
71        }
72        lines.push(Line::from(vec![
73            Span::styled(
74                format!("  {key:>18}  "),
75                Style::default().fg(s::ACCENT_COLOR),
76            ),
77            Span::styled(desc.to_string(), s::normal_style()),
78        ]));
79    }
80
81    lines.push(Line::from(""));
82    lines.push(Line::from(Span::styled(
83        "  Press ? or Esc to close",
84        s::dim_style(),
85    )));
86
87    let help = Paragraph::new(lines).block(
88        Block::default()
89            .title(" Help ")
90            .title_style(
91                Style::default()
92                    .fg(s::ACCENT_COLOR)
93                    .add_modifier(Modifier::BOLD),
94            )
95            .borders(Borders::ALL)
96            .border_style(Style::default().fg(s::BORDER_COLOR))
97            .style(Style::default().bg(s::HEADER_BG)),
98    );
99
100    f.render_widget(help, popup_area);
101}

103fn centered_rect(percent_x: u16, percent_y: u16, area: Rect) -> Rect {
104    let vertical = Layout::vertical([Constraint::Percentage(percent_y)])
105        .flex(Flex::Center)
106        .split(area);
107    Layout::horizontal([Constraint::Percentage(percent_x)])
108        .flex(Flex::Center)
109        .split(vertical[0])[0]
110}
