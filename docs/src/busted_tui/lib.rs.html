16/// Launch the terminal dashboard.
17pub async fn run_tui(config: TuiConfig) -> anyhow::Result<()> {
18    // Setup terminal
19    ratatui::crossterm::terminal::enable_raw_mode()?;
20    ratatui::crossterm::execute!(
21        std::io::stdout(),
22        ratatui::crossterm::terminal::EnterAlternateScreen,
23        ratatui::crossterm::event::EnableMouseCapture,
24    )?;
25
26    let backend = ratatui::backend::CrosstermBackend::new(std::io::stdout());
27    let mut terminal = ratatui::Terminal::new(backend)?;
28
29    let result = run_app(&mut terminal, config).await;
30
31    // Restore terminal
32    ratatui::crossterm::terminal::disable_raw_mode()?;
33    ratatui::crossterm::execute!(
34        std::io::stdout(),
35        ratatui::crossterm::terminal::LeaveAlternateScreen,
36        ratatui::crossterm::event::DisableMouseCapture,
37    )?;
38    terminal.show_cursor()?;
39
40    result
41}

43async fn run_app(
44    terminal: &mut ratatui::Terminal<ratatui::backend::CrosstermBackend<std::io::Stdout>>,
45    config: TuiConfig,
46) -> anyhow::Result<()> {
47    let mut app = App::new(config.demo_mode);
48    let (mut events, busted_tx) = EventHandler::new_with_busted_tx(1000);
49
50    // Spawn data source
51    if config.demo_mode {
52        tokio::spawn(demo::start(busted_tx));
53    } else {
54        tokio::spawn(socket::unix_socket_bridge(busted_tx));
55    }
56
57    // Initial draw
58    terminal.draw(|f| ui::draw(f, &mut app))?;
59
60    while app.running {
61        if let Some(event) = events.next().await {
62            match event {
63                AppEvent::Key(key) => handle_key(&mut app, key),
64                AppEvent::Busted(ev) => app.push_event(*ev),
65                AppEvent::Tick => app.tick(),
66                AppEvent::Resize(_, _) => {} // redraw handles this
67            }
68            terminal.draw(|f| ui::draw(f, &mut app))?;
69        }
70    }
71
72    Ok(())
73}

75fn handle_key(app: &mut App, key: ratatui::crossterm::event::KeyEvent) {
76    let ctrl = key.modifiers.contains(KeyModifiers::CONTROL);
77
78    // Ctrl-C always quits
79    if ctrl && key.code == KeyCode::Char('c') {
80        app.running = false;
81        return;
82    }
83
84    // Help popup intercepts all keys
85    if app.show_help {
86        match key.code {
87            KeyCode::Char('?') | KeyCode::Esc | KeyCode::Char('q') => {
88                app.show_help = false;
89            }
90            _ => {}
91        }
92        return;
93    }
94
95    // Search mode — only text input keys
96    if app.input_mode == InputMode::Search {
97        match key.code {
98            KeyCode::Esc => app.input_mode = InputMode::Normal,
99            KeyCode::Enter => app.input_mode = InputMode::Normal,
100            KeyCode::Backspace => app.handle_search_backspace(),
101            KeyCode::Char(c) => app.handle_search_input(c),
102            _ => {}
103        }
104        return;
105    }
106
107    // Detail panel mode — j/k scroll detail, n/N navigate events
108    if app.show_detail && app.active_tab == Tab::Events {
109        match key.code {
110            KeyCode::Char('q') => {
111                app.running = false;
112                return;
113            }
114            KeyCode::Esc => {
115                app.show_detail = false;
116                return;
117            }
118            KeyCode::Char('?') => {
119                app.show_help = true;
120                return;
121            }
122            // Scroll detail content
123            KeyCode::Char('j') | KeyCode::Down => {
124                app.detail_scroll_down();
125                return;
126            }
127            KeyCode::Char('k') | KeyCode::Up => {
128                app.detail_scroll_up();
129                return;
130            }
131            KeyCode::Char('d') if ctrl => {
132                app.detail_half_page_down();
133                return;
134            }
135            KeyCode::Char('u') if ctrl => {
136                app.detail_half_page_up();
137                return;
138            }
139            KeyCode::Char('f') if ctrl => {
140                app.detail_full_page_down();
141                return;
142            }
143            KeyCode::Char('b') if ctrl => {
144                app.detail_full_page_up();
145                return;
146            }
147            KeyCode::PageDown => {
148                app.detail_full_page_down();
149                return;
150            }
151            KeyCode::PageUp => {
152                app.detail_full_page_up();
153                return;
154            }
155            KeyCode::Char('g') | KeyCode::Home => {
156                app.detail_scroll = 0;
157                return;
158            }
159            KeyCode::Char('G') | KeyCode::End => {
160                // Scroll to a large value; the Paragraph widget clamps internally.
161                app.detail_scroll = usize::MAX / 2;
162                return;
163            }
164            // Navigate events while detail is open
165            KeyCode::Char('n') => {
166                app.select_next_event_in_detail();
167                return;
168            }
169            KeyCode::Char('N') => {
170                app.select_prev_event_in_detail();
171                return;
172            }
173            // Let through keys that make sense globally: tabs, pause, etc.
174            _ => {} // fall through
175        }
176    }
177
178    // Normal mode
179    match key.code {
180        KeyCode::Char('q') => app.running = false,
181
182        // Vim movement
183        KeyCode::Char('j') | KeyCode::Down => app.select_next(),
184        KeyCode::Char('k') | KeyCode::Up => app.select_prev(),
185        KeyCode::Char('g') | KeyCode::Home => app.select_first(),
186        KeyCode::Char('G') | KeyCode::End => app.select_last(),
187
188        // Vim page scroll
189        KeyCode::Char('d') if ctrl => app.half_page_down(),
190        KeyCode::Char('u') if ctrl => app.half_page_up(),
191        KeyCode::Char('f') if ctrl => app.full_page_down(),
192        KeyCode::Char('b') if ctrl => app.full_page_up(),
193        KeyCode::PageDown => app.full_page_down(),
194        KeyCode::PageUp => app.full_page_up(),
195
196        // Vim viewport jumps
197        KeyCode::Char('H') => app.select_high(),
198        KeyCode::Char('M') => app.select_mid(),
199        KeyCode::Char('L') => app.select_low(),
200
201        // Actions
202        KeyCode::Enter => app.toggle_detail(),
203        KeyCode::Esc => {
204            if !app.search_query.is_empty() {
205                app.search_query.clear();
206            } else if app.show_detail {
207                app.show_detail = false;
208            }
209        }
210        KeyCode::Char(' ') => app.toggle_pause(),
211        KeyCode::Char('/') => {
212            app.input_mode = InputMode::Search;
213        }
214
215        // Tabs
216        KeyCode::Tab => app.active_tab = app.active_tab.next(),
217        KeyCode::BackTab => app.active_tab = app.active_tab.prev(),
218        KeyCode::Char('1') => app.active_tab = Tab::Events,
219        KeyCode::Char('2') => app.active_tab = Tab::Processes,
220        KeyCode::Char('3') => app.active_tab = Tab::Providers,
221
222        // Filters & actions
223        KeyCode::Char('c') => app.clear_events(),
224        KeyCode::Char('C') => app.clear_filters(),
225        KeyCode::Char('p') => app.cycle_provider_filter(),
226        KeyCode::Char('t') => app.cycle_type_filter(),
227        KeyCode::Char('?') => app.show_help = true,
228        _ => {}
229    }
230}
