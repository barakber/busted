204fn now_timestamp() -> String {
205    let dur = SystemTime::now()
206        .duration_since(UNIX_EPOCH)
207        .unwrap_or_default();
208    let total_secs = dur.as_secs();
209    let hours = (total_secs / 3600) % 24;
210    let minutes = (total_secs / 60) % 60;
211    let seconds = total_secs % 60;
212    let millis = dur.subsec_millis();
213    format!("{hours:02}:{minutes:02}:{seconds:02}.{millis:03}")
214}

216fn make_process(scenario: &Scenario) -> ProcessInfo {
217    ProcessInfo {
218        pid: scenario.pid,

229fn make_identity(scenario: &Scenario) -> Option<IdentityInfo> {
230    scenario
231        .identity_narrative
232        .as_ref()
233        .map(|narrative| IdentityInfo {
234            id: scenario.pid as u64 * 0x1234567,
235            instance: format!("pid:{}", scenario.pid),
236            confidence: 0.85,
237            match_type: Some("CompositeMatch(0.85)".into()),
238            narrative: Some(narrative.to_string()),
239            timeline: None,
240            timeline_len: Some(5),
241            prompt_fingerprint: None,
242            behavioral_digest: None,
243            capability_hash: None,
244            graph_node_count: Some(12),
245            graph_edge_count: Some(18),
246        })
247}

253fn make_connect(scenario: &Scenario) -> BustedEvent {
254    BustedEvent {
255        timestamp: now_timestamp(),
256        process: make_process(scenario),
257        session_id: session_id(scenario),
258        identity: None,
259        policy: None,
260        action: AgenticAction::Network {
261            kind: NetworkEventKind::Connect,
262            src_ip: "192.168.1.100".into(),
263            src_port: 40000 + (scenario.pid as u16 % 10000),
264            dst_ip: scenario.dst_ip.into(),
265            dst_port: scenario.dst_port,
266            bytes: 0,
267            sni: Some(scenario.sni.into()),
268            provider: Some(scenario.provider.into()),
269        },
270    }
271}

436pub async fn start(tx: mpsc::UnboundedSender<AppEvent>) {
437    let mut cycle = 0usize;
438    loop {
439        let scenario = &SCENARIOS[cycle % SCENARIOS.len()];
440
441        // 1. Connect
442        let _ = tx.send(AppEvent::Busted(Box::new(make_connect(scenario))));
443        sleep(Duration::from_millis(50)).await;
444
445        // 2. MCP or Prompt
446        if scenario.mcp_method.is_some() {
447            let _ = tx.send(AppEvent::Busted(Box::new(make_mcp_request(scenario))));
448            sleep(Duration::from_millis(100)).await;
449            let _ = tx.send(AppEvent::Busted(Box::new(make_mcp_response(scenario))));
450        } else {
451            let request_bytes = 256 + (cycle as u64 * 37) % 512;
452            let _ = tx.send(AppEvent::Busted(Box::new(make_prompt(
453                scenario,
454                request_bytes,
455            ))));
456            sleep(Duration::from_millis(100)).await;
457
458            // 3. Optional tool call
459            if scenario.tool_name.is_some() {
460                let _ = tx.send(AppEvent::Busted(Box::new(make_tool_call(scenario))));
461                sleep(Duration::from_millis(80)).await;
462            }
463
464            // 4. PII detection
465            if scenario.pii {
466                let _ = tx.send(AppEvent::Busted(Box::new(make_pii_detected(scenario))));
467                sleep(Duration::from_millis(50)).await;
468            }
469
470            // 5. Response
471            let response_bytes = 1024 + (cycle as u64 * 73) % 4096;
472            let _ = tx.send(AppEvent::Busted(Box::new(make_response(
473                scenario,
474                response_bytes,
475            ))));
476        }
477
478        // Occasional file-access events (every 3rd cycle)
479        if cycle % 3 == 0 {
480            let _ = tx.send(AppEvent::Busted(Box::new(make_file_access(
481                "claude",
482                9901,
483                "/home/user/.claude/settings.json",
484                "read",
485                "path_pattern:.claude",
486            ))));
487            sleep(Duration::from_millis(30)).await;
488
489            // File data: show content read from settings
490            let _ = tx.send(AppEvent::Busted(Box::new(make_file_data(
491                "claude",
492                9901,
493                "/home/user/.claude/settings.json",
494                "read",
495                r#"{"theme":"dark","model":"claude-sonnet-4-5-20250929","permissions":{"allow_network":true}}"#,
496            ))));
497            sleep(Duration::from_millis(30)).await;
498        }
499        if cycle % 5 == 0 {
500            let _ = tx.send(AppEvent::Busted(Box::new(make_file_access(
501                "cursor",
502                9902,
503                "/home/user/project/CLAUDE.md",
504                "read",
505                "path_pattern:CLAUDE.md",
506            ))));
507            sleep(Duration::from_millis(30)).await;
508        }
509
510        cycle += 1;
511
512        // 300-500ms between scenarios
513        let delay = 300 + (cycle * 29) % 200;
514        sleep(Duration::from_millis(delay as u64)).await;
515    }
