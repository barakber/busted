1//! Candle-based embedding backend.
2//!
3//! Loads `sentence-transformers/all-MiniLM-L6-v2` via HuggingFace Hub,
4//! tokenizes input, runs through BertModel, applies mean pooling + L2 norm.
5
6use anyhow::{Context, Result};
7use candle_core::{DType, Device, Tensor};
8use candle_nn::VarBuilder;
9use candle_transformers::models::bert::{BertModel, Config, DTYPE};
10use hf_hub::{api::sync::Api, Repo, RepoType};
11use tokenizers::Tokenizer;
12
13use crate::{EMBEDDING_DIM, MODEL_ID, MODEL_REVISION};
14
15pub struct CandleEmbedder {
16    model: BertModel,
17    tokenizer: Tokenizer,
18    device: Device,
19}
20
21impl CandleEmbedder {
22    pub fn new() -> Result<Self> {
23        log::info!("Loading embedding model: {MODEL_ID}");
24
25        let repo = Repo::with_revision(
26            MODEL_ID.to_string(),
27            RepoType::Model,
28            MODEL_REVISION.to_string(),
29        );
30        let api = Api::new().context("Failed to initialize HuggingFace Hub API")?;
31        let api_repo = api.repo(repo);
32
33        let config_path = api_repo
34            .get("config.json")
35            .context("Failed to download config.json")?;
36        let tokenizer_path = api_repo
37            .get("tokenizer.json")
38            .context("Failed to download tokenizer.json")?;
39        let weights_path = api_repo
40            .get("model.safetensors")
41            .context("Failed to download model.safetensors")?;
42
43        let config: Config = {
44            let data =
45                std::fs::read_to_string(&config_path).context("Failed to read config.json")?;
46            serde_json::from_str(&data).context("Failed to parse config.json")?
47        };
48
49        let tokenizer = Tokenizer::from_file(&tokenizer_path)
50            .map_err(|e| anyhow::anyhow!("Failed to load tokenizer: {e}"))?;
51
52        let device = Device::Cpu;
53
54        let vb = unsafe {
55            VarBuilder::from_mmaped_safetensors(&[weights_path], DTYPE, &device)
56                .context("Failed to load model weights")?
57        };
58
59        let model = BertModel::load(vb, &config).context("Failed to build BertModel")?;
60
61        log::info!("Embedding model loaded ({EMBEDDING_DIM}-dim)");
62
63        Ok(Self {
64            model,
65            tokenizer,
66            device,
67        })
68    }
69
70    pub fn embed(&mut self, text: &str) -> Result<Vec<f32>> {
71        let mut batch = self.embed_batch(&[text])?;
72        Ok(batch.remove(0))
73    }
74
75    pub fn embed_batch(&mut self, texts: &[&str]) -> Result<Vec<Vec<f32>>> {
76        if texts.is_empty() {
77            return Ok(Vec::new());
78        }
79
80        // Tokenize with padding
81        let encodings = self
82            .tokenizer
83            .encode_batch(texts.to_vec(), true)
84            .map_err(|e| anyhow::anyhow!("Tokenization failed: {e}"))?;
85
86        let max_len = encodings
87            .iter()
88            .map(|e| e.get_ids().len())
89            .max()
90            .unwrap_or(0);
91
92        let mut all_input_ids = Vec::new();
93        let mut all_type_ids = Vec::new();
94        let mut all_attention_mask = Vec::new();
95
96        for encoding in &encodings {
97            let ids = encoding.get_ids();
98            let type_ids = encoding.get_type_ids();
99            let attention = encoding.get_attention_mask();
100
101            // Pad to max_len
102            let mut padded_ids = ids.to_vec();
103            let mut padded_types = type_ids.to_vec();
104            let mut padded_mask = attention.to_vec();
105
106            padded_ids.resize(max_len, 0);
107            padded_types.resize(max_len, 0);
108            padded_mask.resize(max_len, 0);
109
110            all_input_ids.extend(padded_ids);
111            all_type_ids.extend(padded_types);
112            all_attention_mask.extend(padded_mask);
113        }
114
115        let batch_size = texts.len();
116        let input_ids = Tensor::from_vec(all_input_ids, (batch_size, max_len), &self.device)?;
117        let type_ids = Tensor::from_vec(all_type_ids, (batch_size, max_len), &self.device)?;
118        let attention_mask_u32 =
119            Tensor::from_vec(all_attention_mask, (batch_size, max_len), &self.device)?;
120
121        let input_ids = input_ids.to_dtype(DType::U32)?;
122        let type_ids = type_ids.to_dtype(DType::U32)?;
123        let attention_mask = attention_mask_u32.to_dtype(DTYPE)?;
124
125        // Forward pass
126        let output = self
127            .model
128            .forward(&input_ids, &type_ids, Some(&attention_mask))?;
129
130        // Mean pooling: sum(output * mask) / sum(mask)
131        let mask_expanded = attention_mask.unsqueeze(2)?.broadcast_as(output.shape())?;
132        let masked = (output * mask_expanded)?;
133
134        let summed = masked.sum(1)?;
135        let mask_sum = attention_mask.sum(1)?.unsqueeze(1)?;
136        let mask_sum = mask_sum.broadcast_as(summed.shape())?;
137        let pooled = (summed / mask_sum)?;
138
139        // L2 normalize
140        let norms = pooled
141            .sqr()?
142            .sum_keepdim(1)?
143            .sqrt()?
144            .broadcast_as(pooled.shape())?;
145        let normalized = (pooled / norms)?;
146
147        // Extract as Vec<Vec<f32>>
148        let normalized = normalized.to_dtype(DType::F32)?;
149        let flat: Vec<f32> = normalized.flatten_all()?.to_vec1()?;
150        let dim = EMBEDDING_DIM;
151
152        let result: Vec<Vec<f32>> = (0..batch_size)
153            .map(|i| flat[i * dim..(i + 1) * dim].to_vec())
154            .collect();
155
156        Ok(result)
157    }
158}
