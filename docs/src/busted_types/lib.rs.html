1//! The data contract between kernel and userspace for
2//! [Busted](https://github.com/barakber/busted), the eBPF-based LLM/AI communication
3//! monitor.
4//!
5//! When an eBPF probe fires inside the kernel — a TCP connection closes, an SSL handshake
6//! begins, a decrypted payload is captured — it needs to hand that event to userspace. The
7//! two sides share no heap, no allocator, and no serde. They share *these structs*: flat,
8//! `#[repr(C)]`, fixed-size types that the eBPF verifier can prove safe and that Rust can
9//! read directly out of a ring buffer with zero copies.
10//!
11//! # Why this crate exists
12//!
13//! eBPF programs run under extreme constraints: a 512-byte stack, no dynamic allocation,
14//! and a static verifier that rejects any code it can't prove terminates and stays in
15//! bounds. That means no `String`, no `Vec`, no trait objects. Every field must be a
16//! primitive or a fixed-length array, and every struct must be `Copy` and `#[repr(C)]` so
17//! the kernel and userspace agree on layout byte-for-byte.
18//!
19//! At the same time, the userspace agent needs to enrich these raw events with container
20//! metadata, classifier results, ML predictions, and human-readable formatting before
21//! forwarding them to the UI and SIEM sinks. Cramming all of that into `no_std` types
22//! would be painful and pointless.
23//!
24//! `busted-types` solves this by serving two audiences from one crate:
25//!
26//! - **eBPF side** (`no_std`, default) — just the wire types, nothing else.
27//! - **Userspace side** (`user` feature) — adds `aya::Pod` impls, serde, IP-to-string
28//!   helpers, and the enriched [`processed::ProcessedEvent`] that the rest of the system
29//!   consumes.
30//!
31//! # How data flows
32//!
33//! ```text
34//! ┌──────────────┐     #[repr(C)]      ┌──────────────┐     ProcessedEvent     ┌─────────┐
35//! │  eBPF probes │ ──── structs ──────▶ │ busted-agent │ ──── (NDJSON) ──────▶ │ UI/SIEM │
36//! │  (kernel)    │     via RingBuf      │ (userspace)  │     via Unix socket   │         │
37//! └──────────────┘                      └──────────────┘                       └─────────┘
38//! ```
39//!
40//! The eBPF probes write [`NetworkEvent`], [`TlsHandshakeEvent`], or [`TlsDataEvent`]
41//! into a shared ring buffer. The agent reads them out (using aya's `Pod` trait, enabled
42//! by the `user` feature), enriches them into a [`processed::ProcessedEvent`], and
43//! serializes that as NDJSON over a Unix socket.
44//!
45//! # Feature Flags
46//!
47//! - **`user`** — Enables userspace-only functionality:
48//!   - [`aya::Pod`] trait implementations for all event types (required by aya map APIs)
49//!   - [`serde::Serialize`] / [`serde::Deserialize`] on [`processed::ProcessedEvent`]
50//!   - Helper methods for IP address conversion, string extraction, etc.
51//!     (in the [`userspace`] module)
52//!
53//! # Core Types
54//!
55//! | Type | Description |
56//! |------|-------------|
57//! | [`NetworkEvent`] | TCP/UDP network event captured by kprobes |
58//! | [`TlsHandshakeEvent`] | TLS SNI extracted from `SSL_ctrl` uprobe |
59//! | [`TlsDataEvent`] | Decrypted TLS payload from `SSL_write`/`SSL_read` uprobes |
60//! | [`TlsConnKey`] | Composite key `(pid, ssl_ptr)` for the TLS verdict map |
61//! | [`AgentIdentity`] | Identity record for processes communicating with LLM providers |
62//! | [`processed::ProcessedEvent`] | Enriched event for UI/SIEM consumption (requires `user` feature) |

75/// Event types for LLM/AI communication monitoring
76#[repr(u8)]
77#[derive(Clone, Copy, Debug)]
78pub enum EventType {
79    /// TCP connection attempt
80    TcpConnect = 1,
81    /// Data sent
82    DataSent = 2,
83    /// Data received
84    DataReceived = 3,
85    /// Connection closed
86    ConnectionClosed = 4,
87    /// DNS query (UDP to port 53)
88    DnsQuery = 5,
89    /// TLS handshake (SNI extraction)
90    TlsHandshake = 6,
91    /// TLS plaintext data write (SSL_write capture)
92    TlsDataWrite = 7,
93    /// TLS plaintext data read (SSL_read capture)
94    TlsDataRead = 8,
95}

190/// TLS plaintext data captured by eBPF uprobes on SSL_write/SSL_read
191#[repr(C)]
192#[derive(Clone, Copy)]
193pub struct TlsDataEvent {
194    /// Event type: 7=write, 8=read (at byte offset 0 for dispatch)
195    pub event_type: u8,
196    /// Direction: 0=write, 1=read
197    pub direction: u8,
198    /// Actual bytes captured (may be less than TLS_PAYLOAD_MAX)
199    pub payload_len: u16,
200    /// Process ID
201    pub pid: u32,
202    /// Thread ID
203    pub tid: u32,
204    pub _pad: u32,
205    /// SSL struct pointer — unique connection identifier

207    /// Timestamp (nanoseconds since boot)

209    /// Process/command name

211    /// 1 if this is the first data on this connection
212    pub is_first_chunk: u8,
213    pub _pad2: [u8; 3],
214    /// Captured plaintext payload
215    pub payload: [u8; TLS_PAYLOAD_MAX],
216}

397    /// Enriched event ready for UI display, SIEM export, and policy evaluation.
398    ///
399    /// Produced by the agent from raw eBPF events after classification, ML analysis,
400    /// and container/Kubernetes enrichment. Serialized as NDJSON over the Unix socket
401    /// to the UI and SIEM sinks.
402    #[derive(Clone, Debug, Serialize, Deserialize)]
403    pub struct ProcessedEvent {
404        /// Event type string (e.g. `"TCP_CONNECT"`, `"TLS_DATA_WRITE"`).
405        pub event_type: String,
406        /// Human-readable timestamp (`HH:MM:SS.mmm`).
407        pub timestamp: String,
408        /// Process ID.
409        pub pid: u32,
410        /// User ID.
411        pub uid: u32,
412        /// Process/command name.
413        pub process_name: String,
414        /// Source IP address (string).
415        pub src_ip: String,
416        /// Source port.
417        pub src_port: u16,
418        /// Destination IP address (string).
419        pub dst_ip: String,
420        /// Destination port.
421        pub dst_port: u16,
422        /// Bytes transferred.
423        pub bytes: u64,
424        /// LLM provider name from IP/SNI classification.
425        pub provider: Option<String>,
426        /// Policy decision (`"allow"`, `"audit"`, `"deny"`).
427        pub policy: Option<String>,
428        /// Short container ID (first 12 hex chars).
429        pub container_id: String,
430        /// Cgroup ID from the kernel.
431        #[serde(default)]
432        pub cgroup_id: u64,
433        /// Requests per second for this PID.
434        #[serde(default)]
435        pub request_rate: Option<f64>,
436        /// Cumulative bytes for this PID's session.
437        #[serde(default)]
438        pub session_bytes: Option<u64>,
439        /// Kubernetes pod name.
440        #[serde(default)]
441        pub pod_name: Option<String>,
442        /// Kubernetes namespace.
443        #[serde(default)]
444        pub pod_namespace: Option<String>,
445        /// Kubernetes service account.
446        #[serde(default)]
447        pub service_account: Option<String>,
448        /// ML classifier confidence (0.0-1.0).
449        #[serde(default, skip_serializing_if = "Option::is_none")]
450        pub ml_confidence: Option<f64>,
451        /// ML-predicted provider name.
452        #[serde(default, skip_serializing_if = "Option::is_none")]
453        pub ml_provider: Option<String>,
454        /// ML behavioral class (e.g. `"LlmApi(OpenAI)"`).

457        /// HDBSCAN cluster ID (-1 = noise).

460        /// TLS SNI hostname.
461        #[serde(default, skip_serializing_if = "Option::is_none")]
462        pub sni: Option<String>,
463        /// TLS content class (e.g. `"LlmApi"`, `"Mcp"`).

466        /// TLS classification details (provider, endpoint, model).

469        /// Decrypted TLS payload (lossy UTF-8).
470        #[serde(default, skip_serializing_if = "Option::is_none")]
471        pub tls_payload: Option<String>,
472        /// Content class from `busted-classifier`.

484        /// MCP method name.
485        #[serde(default, skip_serializing_if = "Option::is_none")]
486        pub mcp_method: Option<String>,
487        /// MCP method category.
488        #[serde(default, skip_serializing_if = "Option::is_none")]
489        pub mcp_category: Option<String>,
490        /// SDK/agent name and version.
491        #[serde(default, skip_serializing_if = "Option::is_none")]
492        pub agent_sdk: Option<String>,
493        /// Behavioral signature hash.
494        #[serde(default, skip_serializing_if = "Option::is_none")]
495        pub agent_fingerprint: Option<u64>,
496        /// Content classifier confidence (0.0-1.0).
497        #[serde(default, skip_serializing_if = "Option::is_none")]

499        /// Whether PII was detected in the payload.

502    }

505/// Userspace helper methods for eBPF event types (requires `user` feature).
506#[cfg(feature = "user")]
507pub mod userspace {
508    use super::*;
509    use std::net::{IpAddr, Ipv4Addr, Ipv6Addr};
510
511    impl TlsHandshakeEvent {
512        /// Get SNI hostname as string
513        pub fn sni_str(&self) -> &str {
514            let len = self
515                .sni
516                .iter()
517                .position(|&c| c == 0)
518                .unwrap_or(self.sni.len());
519            std::str::from_utf8(&self.sni[..len]).unwrap_or("<invalid>")
520        }
521
522        /// Get process name as string
523        pub fn process_name(&self) -> &str {
524            let len = self
525                .comm
526                .iter()
527                .position(|&c| c == 0)
528                .unwrap_or(self.comm.len());
529            std::str::from_utf8(&self.comm[..len]).unwrap_or("<invalid>")
530        }
531    }
532
533    impl TlsDataEvent {
534        /// Get the captured payload bytes (up to payload_len)
535        pub fn payload_bytes(&self) -> &[u8] {
536            let len = (self.payload_len as usize).min(crate::TLS_PAYLOAD_MAX);
537            &self.payload[..len]
538        }
539
540        /// Get process name as string
541        pub fn process_name(&self) -> &str {
542            let len = self
543                .comm
544                .iter()
545                .position(|&c| c == 0)
546                .unwrap_or(self.comm.len());
547            std::str::from_utf8(&self.comm[..len]).unwrap_or("<invalid>")
548        }
549    }
550
551    impl NetworkEvent {
552        /// Get source IP as std::net::IpAddr
553        pub fn source_ip(&self) -> IpAddr {
554            match self.family {
555                2 => IpAddr::V4(Ipv4Addr::from(u32::from_be(unsafe { self.saddr.ipv4 }))),
556                10 => IpAddr::V6(Ipv6Addr::from(unsafe { self.saddr.ipv6 })),
557                _ => IpAddr::V4(Ipv4Addr::from(0)),
558            }
559        }
560
561        /// Get destination IP as std::net::IpAddr
562        pub fn dest_ip(&self) -> IpAddr {
563            match self.family {
564                2 => IpAddr::V4(Ipv4Addr::from(u32::from_be(unsafe { self.daddr.ipv4 }))),
565                10 => IpAddr::V6(Ipv6Addr::from(unsafe { self.daddr.ipv6 })),
566                _ => IpAddr::V4(Ipv4Addr::from(0)),
567            }
568        }
569
570        /// Get process name as string
571        pub fn process_name(&self) -> &str {
572            let len = self
573                .comm
574                .iter()
575                .position(|&c| c == 0)
576                .unwrap_or(self.comm.len());
577            std::str::from_utf8(&self.comm[..len]).unwrap_or("<invalid>")
578        }
579
580        /// Get container ID as string
581        pub fn container_id_str(&self) -> &str {
582            let len = self
583                .container_id
584                .iter()
585                .position(|&c| c == 0)
586                .unwrap_or(self.container_id.len());
587            std::str::from_utf8(&self.container_id[..len]).unwrap_or("")
588        }
589    }
