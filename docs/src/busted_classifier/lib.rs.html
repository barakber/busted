1//! Tells you whether a decrypted TLS payload is an LLM API call, an MCP message, or
2//! just normal HTTP — in microseconds, from a single buffer, with no connection state.
3//!
4//! # The problem
5//!
6//! Every major LLM provider (OpenAI, Anthropic, Google, AWS Bedrock, Cohere, Mistral,
7//! Groq, and more) serves its API over HTTPS. To a traditional network monitor, a
8//! `POST /v1/chat/completions` to `api.openai.com` looks identical to any other TLS
9//! connection — just encrypted bytes on port 443. Even after you've intercepted the
10//! plaintext (via eBPF uprobes on `SSL_write`/`SSL_read`), you're left with a raw byte
11//! buffer. Is it a GPT-4 request? A Claude conversation? An MCP tool invocation? Or
12//! someone browsing Reddit? This crate answers that question.
13//!
14//! `busted-classifier` encodes deep knowledge of LLM API shapes, MCP's JSON-RPC 2.0
15//! protocol, SDK user-agent strings, and streaming response formats into a fast,
16//! stateless classification pipeline. Hand it a `&[u8]` and a direction, and it tells
17//! you exactly what you're looking at.
18//!
19//! # How it works
20//!
21//! Classification flows through four layers, each adding detail:
22//!
23//! First, we figure out if the payload is even HTTP. The [`http`] module uses [`nom`]
24//! combinators to parse HTTP/1.1 request and response headers, detect HTTP/2 binary
25//! framing, and identify Server-Sent Events streams. If we can't parse it as HTTP, we
26//! still try — the payload might be a raw JSON body without framing.
27//!
28//! Next, we check whether the HTTP endpoint matches a known LLM provider. The [`llm`]
29//! module maintains a registry of API paths and host patterns — `/v1/chat/completions`
30//! on `api.openai.com`, `/v1/messages` on `api.anthropic.com`, and dozens more. It also
31//! inspects JSON response bodies for telltale fields like `choices`, `model`, and
32//! `completion`. The [`mcp`] module separately detects MCP JSON-RPC 2.0 methods
33//! (`tools/call`, `resources/read`, etc.) and categorizes them.
34//!
35//! Then we fingerprint the SDK. The [`fingerprint`] module extracts the `User-Agent`
36//! header (`openai-python/1.12.0`, `anthropic-typescript/0.19.0`), pulls model
37//! parameters from the JSON body, and computes a behavioral signature hash that can
38//! identify the same agent across requests.
39//!
40//! Finally, if the `pii` feature is enabled, the [`pii`] module scans the payload for
41//! leaked personally identifiable information — email addresses, credit card numbers,
42//! SSNs, phone numbers, and API keys — so you know when sensitive data is being sent
43//! to an LLM.
44//!
45//! # Architecture
46//!
47//! The four layers map to these modules:
48//!
49//! 1. **Protocol detection** — HTTP/1.1 parsing (via [`nom`]), HTTP/2 binary framing,
50//!    SSE stream identification.
51//! 2. **Content classification** — LLM API endpoint matching, MCP JSON-RPC detection.
52//! 3. **Agent fingerprinting** — SDK extraction, model params, behavioral signature.
53//! 4. **PII detection** — Regex scanning for sensitive data patterns.
54//!
55//! # Integration with Busted
56//!
57//! In the Busted pipeline, `busted-agent` captures decrypted TLS payloads via eBPF
58//! uprobes, then calls [`classify()`] on the first chunk of each connection. The
59//! resulting [`Classification`] is folded into a
60//! `ProcessedEvent` and forwarded to the UI
61//! and SIEM sinks. The crate has no dependency on eBPF or aya — it's a pure Rust
62//! library that works anywhere you have bytes to classify.
63//!
64//! # Usage
65//!
66//! ```
67//! use busted_classifier::{classify, Direction};
68//!
69//! let payload = b"POST /v1/chat/completions HTTP/1.1\r\n\
70//!     Host: api.openai.com\r\n\
71//!     Content-Type: application/json\r\n\r\n\
72//!     {\"model\":\"gpt-4\",\"messages\":[]}";
73//!
74//! let result = classify(payload, Direction::Write, None);
75//! assert!(result.is_interesting);
76//! assert_eq!(result.provider(), Some("OpenAI"));
77//! assert_eq!(result.model(), Some("gpt-4"));
78//! ```
79//!
80//! # Feature Flags
81//!
82//! - **`pii`** — Enables PII detection via regex scanning (adds `regex` dependency).
83//!   When disabled, [`PiiFlags`] fields are always `false`.

106/// Direction of TLS data flow.
107#[derive(Debug, Clone, Copy, PartialEq, Eq)]
108pub enum Direction {
109    /// Outbound data (client to server, e.g. `SSL_write`).
110    Write,
111    /// Inbound data (server to client, e.g. `SSL_read`).
112    Read,
113}
114
115/// Layer 1: Protocol detection result.
116#[derive(Debug, Clone)]
117pub enum Protocol {
118    /// HTTP/1.1 request with parsed headers and body offset.
119    Http1Request(http::HttpRequestInfo),
120    /// HTTP/1.1 response with parsed headers and body offset.
121    Http1Response(http::HttpResponseInfo),
122    /// Server-Sent Events stream (`data: ...` lines).
123    Sse,
124    /// HTTP/2 binary framing (connection preface or frames).
125    Http2Binary,
126    /// Non-text binary data.
127    Binary,
128    /// Could not determine protocol.
129    Unknown,
130}
131
132/// Layer 2: Content classification.
133#[derive(Debug, Clone)]
134pub enum ContentClass {
135    /// LLM API request or response.
136    LlmApi(LlmApiInfo),
137    /// MCP JSON-RPC 2.0 message.
138    Mcp(McpInfo),
139    /// LLM streaming response (SSE).
140    LlmStream(LlmStreamInfo),
141    /// Generic JSON-RPC (non-MCP).
142    JsonRpc,
143    /// Generic HTTP (not LLM/MCP).
144    GenericHttp,
145}
146
147/// Full classification result.
148#[derive(Debug, Clone)]
149pub struct Classification {
150    /// Whether this payload is interesting (LLM/MCP related).
151    pub is_interesting: bool,
152    /// Layer 1: protocol detection.
153    pub protocol: Protocol,
154    /// Layer 2: content classification.
155    pub content: Option<ContentClass>,
156    /// Layer 3: agent/SDK fingerprint.
157    pub fingerprint: Option<AgentFingerprint>,
158    /// PII detection flags (all false when `pii` feature disabled).
159    pub pii_flags: PiiFlags,
160    /// Classification confidence (0.0–1.0).
161    pub confidence: f32,
162}

164/// Classify a decrypted TLS payload chunk.
165///
166/// Stateless per-call — connection tracking stays in busted-agent.
167pub fn classify(payload: &[u8], direction: Direction, sni_hint: Option<&str>) -> Classification {
168    let pii_flags = pii::scan(payload);
169
170    // Layer 1: detect protocol
171    let protocol = detect_protocol(payload, direction);
172
173    // Layer 2 + 3: classify content and extract fingerprint
174    let (content, fp, confidence) = match &protocol {
175        Protocol::Http1Request(req) => classify_http_request(req, payload, sni_hint),
176        Protocol::Http1Response(resp) => classify_http_response(resp, payload, sni_hint),
177        Protocol::Sse => classify_sse(payload, sni_hint),
178        Protocol::Http2Binary => (None, None, 0.3),
179        Protocol::Binary | Protocol::Unknown => classify_raw_body(payload, sni_hint),
180    };
181
182    let is_interesting = content
183        .as_ref()
184        .map(|c| !matches!(c, ContentClass::GenericHttp))
185        .unwrap_or(false);
186
187    Classification {
188        is_interesting,
189        protocol,
190        content,
191        fingerprint: fp,
192        pii_flags,
193        confidence,
194    }
195}

197/// Layer 1: Detect protocol from raw payload bytes.
198fn detect_protocol(payload: &[u8], direction: Direction) -> Protocol {
199    if payload.is_empty() {
200        return Protocol::Unknown;
201    }
202
203    // Check for HTTP/2 binary framing first
204    if http::is_http2_binary(payload) {
205        return Protocol::Http2Binary;
206    }
207
208    // Check for SSE (often in read direction)
209    if direction == Direction::Read {
210        let text = std::str::from_utf8(payload).unwrap_or("");
211        if text.starts_with("data: ") || text.starts_with("event: ") || text.starts_with(": ") {
212            return Protocol::Sse;
213        }
214    }
215
216    // Try HTTP request/response
217    if http::looks_like_http_request(payload) {
218        if let Some(req) = http::parse_request(payload) {
219            return Protocol::Http1Request(req);
220        }
221    }
222
223    if http::looks_like_http_response(payload) {
224        if let Some(resp) = http::parse_response(payload) {
225            return Protocol::Http1Response(resp);
226        }
227    }
228
229    // Check if it's text (potential JSON body without HTTP headers)
230    if payload.iter().all(|&b| b >= 0x09 && b != 0x7f) || std::str::from_utf8(payload).is_ok() {
231        return Protocol::Unknown;
232    }
233
234    Protocol::Binary
235}

237/// Classify an HTTP/1.1 request.
238fn classify_http_request(
239    req: &http::HttpRequestInfo,
240    full_payload: &[u8],
241    sni_hint: Option<&str>,
242) -> (Option<ContentClass>, Option<AgentFingerprint>, f32) {
243    // Parse JSON body if present
244    let json_fields = req.body_offset.and_then(|offset| {
245        if offset < full_payload.len() {
246            Some(json::analyze(&full_payload[offset..]))
247        } else {
248            None
249        }
250    });
251
252    let empty_json = json::JsonFields::default();
253    let jf = json_fields.as_ref().unwrap_or(&empty_json);
254
255    // Check for MCP first (JSON-RPC 2.0 with MCP methods)
256    if let Some(mcp_info) = mcp::classify(jf) {
257        let fp = fingerprint::build_fingerprint(req, jf);
258        return (Some(ContentClass::Mcp(mcp_info)), Some(fp), 0.95);
259    }
260
261    // Check for LLM API endpoint
262    if let Some(llm_info) = llm::match_request_with_body(req, sni_hint, jf) {
263        let fp = fingerprint::build_fingerprint(req, jf);
264        return (Some(ContentClass::LlmApi(llm_info)), Some(fp), 0.9);
265    }
266
267    // Generic JSON-RPC (non-MCP)
268    if jf.jsonrpc.is_some() {
269        let fp = fingerprint::build_fingerprint(req, jf);
270        return (Some(ContentClass::JsonRpc), Some(fp), 0.6);
271    }
272
273    // Check for LLM-like body without matching endpoint
274    let llm_indicators = count_llm_indicators(jf);
275    if llm_indicators >= 2 {
276        let fp = fingerprint::build_fingerprint(req, jf);
277        let confidence = 0.5 + (llm_indicators as f32 * 0.1).min(0.4);
278        let info = LlmApiInfo {
279            provider: "Unknown".to_string(),
280            endpoint: "unknown".to_string(),
281            model: jf.model.clone(),
282            streaming: jf.stream,
283        };
284        return (Some(ContentClass::LlmApi(info)), Some(fp), confidence);
285    }
286
287    // Generic HTTP
288    (Some(ContentClass::GenericHttp), None, 0.3)
289}

291/// Classify an HTTP/1.1 response.
292fn classify_http_response(
293    resp: &http::HttpResponseInfo,
294    full_payload: &[u8],
295    sni_hint: Option<&str>,
296) -> (Option<ContentClass>, Option<AgentFingerprint>, f32) {
297    // Parse JSON body
298    let json_fields = resp.body_offset.and_then(|offset| {
299        if offset < full_payload.len() {
300            Some(json::analyze(&full_payload[offset..]))
301        } else {
302            None
303        }
304    });
305
306    let empty_json = json::JsonFields::default();
307    let jf = json_fields.as_ref().unwrap_or(&empty_json);
308
309    // Check for MCP response
310    if let Some(mcp_info) = mcp::classify(jf) {
311        return (Some(ContentClass::Mcp(mcp_info)), None, 0.9);
312    }
313
314    // Check for LLM response patterns
315    if let Some(llm_info) = llm::classify_response(jf, sni_hint) {
316        return (Some(ContentClass::LlmApi(llm_info)), None, 0.85);
317    }
318
319    // Generic JSON-RPC response
320    if jf.jsonrpc.is_some() {
321        return (Some(ContentClass::JsonRpc), None, 0.6);
322    }
323
324    (Some(ContentClass::GenericHttp), None, 0.3)
325}

327/// Classify SSE streaming data.
328fn classify_sse(
329    payload: &[u8],
330    sni_hint: Option<&str>,
331) -> (Option<ContentClass>, Option<AgentFingerprint>, f32) {
332    let text = String::from_utf8_lossy(payload);
333    if let Some(stream_info) = llm::detect_sse_stream(&text, sni_hint) {
334        return (Some(ContentClass::LlmStream(stream_info)), None, 0.85);
335    }
336    (Some(ContentClass::GenericHttp), None, 0.3)
337}

339/// Classify raw body bytes (no HTTP framing detected).
340fn classify_raw_body(
341    payload: &[u8],
342    sni_hint: Option<&str>,
343) -> (Option<ContentClass>, Option<AgentFingerprint>, f32) {
344    // Try to parse as JSON directly
345    let jf = json::analyze(payload);
346
347    // Check MCP
348    if let Some(mcp_info) = mcp::classify(&jf) {
349        return (Some(ContentClass::Mcp(mcp_info)), None, 0.8);
350    }
351
352    // Check LLM response
353    if let Some(llm_info) = llm::classify_response(&jf, sni_hint) {
354        return (Some(ContentClass::LlmApi(llm_info)), None, 0.7);
355    }
356
357    // Check LLM indicators in body
358    let llm_indicators = count_llm_indicators(&jf);
359    if llm_indicators >= 2 {
360        let confidence = 0.4 + (llm_indicators as f32 * 0.1).min(0.3);
361        let info = LlmApiInfo {
362            provider: "Unknown".to_string(),
363            endpoint: "unknown".to_string(),
364            model: jf.model.clone(),
365            streaming: jf.stream,
366        };
367        return (Some(ContentClass::LlmApi(info)), None, confidence);
368    }
369
370    // Check SSE patterns
371    if let Ok(text) = std::str::from_utf8(payload) {
372        if let Some(stream_info) = llm::detect_sse_stream(text, sni_hint) {
373            return (Some(ContentClass::LlmStream(stream_info)), None, 0.7);
374        }
375    }
376
377    (None, None, 0.0)
378}

380/// Count LLM-indicative fields in JSON.
381fn count_llm_indicators(jf: &json::JsonFields) -> u32 {
382    let mut count = 0u32;
383    if jf.model.is_some() {
384        count += 1;
385    }
386    if jf.has_messages {
387        count += 1;
388    }
389    if jf.has_prompt {
390        count += 1;
391    }
392    if jf.temperature.is_some() {
393        count += 1;
394    }
395    if jf.max_tokens.is_some() {
396        count += 1;
397    }
398    if jf.has_choices {
399        count += 1;
400    }
401    if jf.has_content {
402        count += 1;
403    }
404    if jf.has_completion {
405        count += 1;
406    }
407    count
408}

412impl Classification {
413    /// Short string describing the content class.
414    pub fn content_class_str(&self) -> Option<&'static str> {
415        self.content.as_ref().map(|c| match c {
416            ContentClass::LlmApi(_) => "LlmApi",
417            ContentClass::Mcp(_) => "Mcp",
418            ContentClass::LlmStream(_) => "LlmStream",
419            ContentClass::JsonRpc => "JsonRpc",
420            ContentClass::GenericHttp => "GenericHttp",
421        })
422    }
423
424    /// Provider name if classified as LLM or MCP.
425    pub fn provider(&self) -> Option<&str> {
426        match &self.content {
427            Some(ContentClass::LlmApi(info)) => Some(&info.provider),
428            Some(ContentClass::LlmStream(info)) => Some(&info.provider),
429            _ => None,
430        }
431    }
432
433    /// Endpoint name if classified as LLM API.
434    pub fn endpoint(&self) -> Option<&str> {
435        match &self.content {
436            Some(ContentClass::LlmApi(info)) => Some(&info.endpoint),
437            _ => None,
438        }
439    }
440
441    /// Model name if available.
442    pub fn model(&self) -> Option<&str> {
443        match &self.content {
444            Some(ContentClass::LlmApi(info)) => info.model.as_deref(),
445            _ => self
446                .fingerprint
447                .as_ref()
448                .and_then(|fp| fp.model_params.model.as_deref()),
449        }
450    }
451
452    /// MCP method if classified as MCP.
453    pub fn mcp_method(&self) -> Option<&str> {
454        match &self.content {
455            Some(ContentClass::Mcp(info)) => info.method.as_deref(),
456            _ => None,
457        }
458    }
459
460    /// MCP category string if classified as MCP.
461    pub fn mcp_category_str(&self) -> Option<String> {
462        match &self.content {
463            Some(ContentClass::Mcp(info)) => Some(info.category.to_string()),
464            _ => None,
465        }
466    }
467
468    /// SDK name+version string if fingerprinted.
469    pub fn sdk_string(&self) -> Option<String> {
470        self.fingerprint.as_ref().and_then(|fp| {
471            fp.sdk.as_ref().map(|sdk| match &sdk.version {
472                Some(v) => format!("{}/{}", sdk.name, v),
473                None => sdk.name.clone(),
474            })
475        })
476    }
477
478    /// Signature hash if fingerprinted.
479    pub fn signature_hash(&self) -> Option<u64> {
480        self.fingerprint.as_ref().map(|fp| fp.signature_hash)
481    }
482}
