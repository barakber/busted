10/// Parsed HTTP/1.1 request.
11#[derive(Debug, Clone)]
12pub struct HttpRequestInfo {
13    /// HTTP method (e.g. `GET`, `POST`).
14    pub method: String,
15    /// Request path (e.g. `/v1/chat/completions`).
16    pub path: String,
17    /// HTTP version (e.g. `HTTP/1.1`).
18    pub version: String,
19    /// Headers (keys lowercased, values with sensitive data masked).
20    pub headers: HashMap<String, String>,

52/// Quick check: does this look like an HTTP/2 binary frame?
53pub fn is_http2_binary(data: &[u8]) -> bool {
54    // HTTP/2 connection preface
55    if data.starts_with(b"PRI * HTTP/2.0\r\n\r\nSM\r\n\r\n") {
56        return true;
57    }
58    // HTTP/2 frame: 3-byte length + 1-byte type + 1-byte flags + 4-byte stream ID
59    // Type values 0-9 are defined; length is reasonable
60    if data.len() >= 9 {
61        let len = ((data[0] as u32) << 16) | ((data[1] as u32) << 8) | (data[2] as u32);
62        let frame_type = data[3];
63        // Valid frame types: DATA(0)..CONTINUATION(9)
64        if frame_type <= 9 && len < 16_777_216 && len > 0 {
65            // Heuristic: if the high bit of byte 5-8 is clear (stream ID is 31 bits)
66            if data[5] & 0x80 == 0 {
67                return true;
68            }
69        }
70    }
71    false
72}

74/// Quick check: does this look like the start of an HTTP request?
75pub fn looks_like_http_request(data: &[u8]) -> bool {
76    const METHODS: &[&[u8]] = &[
77        b"GET ",
78        b"POST ",
79        b"PUT ",
80        b"DELETE ",
81        b"PATCH ",
82        b"HEAD ",
83        b"OPTIONS ",
84        b"CONNECT ",
85    ];
86    for m in METHODS {
87        if data.starts_with(m) {
88            return true;
89        }
90    }
91    false
92}

99fn is_token_char(c: u8) -> bool {
100    c.is_ascii_alphanumeric() || b"!#$%&'*+-.^_`|~".contains(&c)
101}

103fn parse_method(input: &[u8]) -> IResult<&[u8], &[u8]> {
104    take_while1(|c: u8| c.is_ascii_uppercase())(input)
105}

107#[allow(clippy::type_complexity)]
108fn parse_request_line(input: &[u8]) -> IResult<&[u8], (&[u8], &[u8], &[u8])> {
109    let (input, method) = parse_method(input)?;
110    let (input, _) = space1(input)?;
111    let (input, path) = take_while1(|c: u8| c != b' ' && c != b'\r' && c != b'\n')(input)?;
112    let (input, _) = space1(input)?;
113    let (input, version) = take_while1(|c: u8| c != b'\r' && c != b'\n')(input)?;
114    let (input, _) = tag(b"\r\n")(input)?;
115    Ok((input, (method, path, version)))
116}

118#[allow(clippy::type_complexity)]
119fn parse_status_line(input: &[u8]) -> IResult<&[u8], (&[u8], u16, &[u8])> {
120    let (input, version) = take_while1(|c: u8| c != b' ' && c != b'\r')(input)?;
121    let (input, _) = space1(input)?;
122    let (input, code_bytes) = take_while1(|c: u8| c.is_ascii_digit())(input)?;
123    let code: u16 = std::str::from_utf8(code_bytes)
124        .ok()
125        .and_then(|s| s.parse().ok())
126        .unwrap_or(0);
127    let (input, _) = opt(space1)(input)?;
128    // Reason phrase is optional and runs to CRLF
129    let (input, reason) =
130        take_while1::<_, _, nom::error::Error<&[u8]>>(|c: u8| c != b'\r' && c != b'\n')(input)
131            .unwrap_or((input, b"" as &[u8]));
132    let (input, _) = tag(b"\r\n")(input)?;
133    Ok((input, (version, code, reason)))
134}

136fn parse_header(input: &[u8]) -> IResult<&[u8], (&[u8], &[u8])> {
137    let (input, name) = take_while1(is_token_char)(input)?;
138    let (input, _) = tuple((char(':'), opt(space1)))(input)?;
139    let (input, value) = take_while1(|c: u8| c != b'\r' && c != b'\n')(input)?;
140    let (input, _) = tag(b"\r\n")(input)?;
141    Ok((input, (name, value)))
142}

144/// Returns (headers, remaining_body, headers_complete).
145/// `headers_complete` is true if the `\r\n\r\n` terminator was found.
146fn parse_headers(mut input: &[u8]) -> (HashMap<String, String>, &[u8], bool) {
147    let mut headers = HashMap::new();
148
149    loop {
150        // Check for end of headers
151        if input.starts_with(b"\r\n") {
152            return (headers, &input[2..], true);
153        }
154        if input.is_empty() {
155            // Truncated before end of headers
156            return (headers, input, false);
157        }
158
159        match parse_header(input) {
160            Ok((rest, (name, value))) => {
161                let name_str = String::from_utf8_lossy(name).to_lowercase();
162                // Only store interesting headers
163                if INTERESTING_HEADERS.contains(&name_str.as_str()) {
164                    let val = String::from_utf8_lossy(value).to_string();
165                    // Mask authorization values for security
166                    let val = if name_str == "authorization" {
167                        mask_auth(&val)
168                    } else if name_str == "x-api-key" {
169                        "[present]".to_string()
170                    } else {
171                        val
172                    };
173                    headers.insert(name_str, val);
174                }
175                input = rest;
176            }
177            Err(_) => {
178                // Truncated or malformed header — stop parsing
179                return (headers, input, false);
180            }
181        }
182    }
183}

185fn mask_auth(val: &str) -> String {
186    if val.len() > 12 {
187        format!("{}...{}", &val[..8], &val[val.len() - 4..])
188    } else {
189        "[redacted]".to_string()
190    }
191}

193/// Try to parse an HTTP/1.1 request from raw bytes.
194/// Handles truncated payloads gracefully.
195pub fn parse_request(data: &[u8]) -> Option<HttpRequestInfo> {
196    let (rest, (method, path, version)) = parse_request_line(data).ok()?;
197    let (headers, body, headers_complete) = parse_headers(rest);
198
199    let body_offset = if headers_complete {
200        Some(data.len() - body.len())
201    } else {
202        None
203    };
204
205    Some(HttpRequestInfo {
206        method: String::from_utf8_lossy(method).to_string(),
207        path: String::from_utf8_lossy(path).to_string(),
208        version: String::from_utf8_lossy(version).to_string(),
209        headers,
210        body_offset,
211    })
212}

214/// Try to parse an HTTP/1.1 response from raw bytes.
215pub fn parse_response(data: &[u8]) -> Option<HttpResponseInfo> {
216    let (rest, (version, status_code, reason)) = parse_status_line(data).ok()?;
217    let (headers, body, headers_complete) = parse_headers(rest);
218
219    let body_offset = if headers_complete {
220        Some(data.len() - body.len())
221    } else {
222        None
223    };
224
225    Some(HttpResponseInfo {
226        version: String::from_utf8_lossy(version).to_string(),
227        status_code,
228        reason: String::from_utf8_lossy(reason).to_string(),
229        headers,
230        body_offset,
231    })
232}
