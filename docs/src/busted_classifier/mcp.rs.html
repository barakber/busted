3/// MCP method category.
4#[derive(Debug, Clone, PartialEq, Eq)]
5pub enum McpCategory {
6    /// Tool invocation methods (`tools/call`, `tools/list`).
7    Tools,
8    /// Resource access methods (`resources/read`, `resources/list`).
9    Resources,
10    /// Prompt template methods (`prompts/get`, `prompts/list`).
11    Prompts,
12    /// Lifecycle methods (`initialize`, `shutdown`, `ping`).
13    Lifecycle,
14    /// Completion/autocomplete methods.
15    Completion,
16    /// Logging methods (`logging/setLevel`).
17    Logging,
18    /// Unrecognized MCP method.
19    Other,
20}
21
22impl std::fmt::Display for McpCategory {
23    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
24        match self {
25            McpCategory::Tools => write!(f, "Tools"),
26            McpCategory::Resources => write!(f, "Resources"),
27            McpCategory::Prompts => write!(f, "Prompts"),
28            McpCategory::Lifecycle => write!(f, "Lifecycle"),
29            McpCategory::Completion => write!(f, "Completion"),
30            McpCategory::Logging => write!(f, "Logging"),
31            McpCategory::Other => write!(f, "Other"),
32        }
33    }
34}

36/// Type of JSON-RPC message.
37#[derive(Debug, Clone, PartialEq, Eq)]
38pub enum McpMsgType {
39    /// JSON-RPC request (has `method` and `id`).
40    Request,
41    /// JSON-RPC notification (has `method`, no `id`).
42    Notification,
43    /// JSON-RPC success response (has `result` and `id`).
44    Response,
45    /// JSON-RPC error response (has `error` and `id`).
46    Error,
47}

49impl std::fmt::Display for McpMsgType {
50    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
51        match self {
52            McpMsgType::Request => write!(f, "Request"),
53            McpMsgType::Notification => write!(f, "Notification"),
54            McpMsgType::Response => write!(f, "Response"),
55            McpMsgType::Error => write!(f, "Error"),
56        }
57    }
58}

60/// MCP classification result.
61#[derive(Debug, Clone)]
62pub struct McpInfo {
63    /// MCP method name (e.g. `"tools/call"`, `"initialize"`).
64    pub method: Option<String>,
65    /// Functional category of the method.
66    pub category: McpCategory,
67    /// JSON-RPC message type.
68    pub msg_type: McpMsgType,
69    /// JSON-RPC message ID (present for requests and responses).
70    pub id: Option<String>,
71}

106/// Try to classify JSON fields as MCP JSON-RPC 2.0.
107/// Returns None if this doesn't look like MCP.
108pub fn classify(fields: &JsonFields) -> Option<McpInfo> {
109    // Must have jsonrpc: "2.0"
110    if fields.jsonrpc.as_deref() != Some("2.0") {
111        return None;
112    }
113
114    let msg_type = determine_msg_type(fields);
115    let method = fields.method.clone();
116    let category = method
117        .as_deref()
118        .and_then(lookup_category)
119        .unwrap_or(McpCategory::Other);
120
121    // For responses/errors without a method, we can still identify them as MCP
122    // if they have the JSON-RPC 2.0 structure
123    let is_mcp = method.as_deref().map(is_mcp_method).unwrap_or(false)
124        || matches!(msg_type, McpMsgType::Response | McpMsgType::Error);
125
126    if !is_mcp {
127        return None;
128    }
129
130    Some(McpInfo {
131        method,
132        category,
133        msg_type,
134        id: fields.id.clone(),
135    })
136}
137
138fn determine_msg_type(fields: &JsonFields) -> McpMsgType {
139    if fields.has_error && fields.id.is_some() {
140        McpMsgType::Error
141    } else if fields.has_result && fields.id.is_some() {
142        McpMsgType::Response
143    } else if fields.method.is_some() && fields.id.is_some() {
144        McpMsgType::Request
145    } else if fields.method.is_some() {
146        McpMsgType::Notification
147    } else {
148        McpMsgType::Response // Default for ambiguous structures
149    }
150}
151
152fn lookup_category(method: &str) -> Option<McpCategory> {
153    for &(m, ref cat) in MCP_METHODS {
154        if m == method {
155            return Some(cat.clone());
156        }
157    }
158    None
159}

161fn is_mcp_method(method: &str) -> bool {
162    MCP_METHODS.iter().any(|&(m, _)| m == method)
163        || method.starts_with("tools/")
164        || method.starts_with("resources/")
165        || method.starts_with("prompts/")
166        || method.starts_with("logging/")
167        || method.starts_with("completion/")
168        || method.starts_with("notifications/")
169        || method.starts_with("sampling/")
170        || method == "initialize"
171        || method == "initialized"
172        || method == "ping"
173        || method == "shutdown"
174        || method == "cancelled"
175}

182    #[test]
183    fn test_tools_call_request() {
184        let body = br#"{"jsonrpc":"2.0","method":"tools/call","params":{"name":"search"},"id":1}"#;
185        let fields = json::analyze(body);
186        let mcp = classify(&fields).unwrap();
187        assert_eq!(mcp.method.as_deref(), Some("tools/call"));
188        assert_eq!(mcp.category, McpCategory::Tools);
189        assert_eq!(mcp.msg_type, McpMsgType::Request);
190    }
191
192    #[test]
193    fn test_initialize_notification() {
194        let body = br#"{"jsonrpc":"2.0","method":"initialized"}"#;
195        let fields = json::analyze(body);
196        let mcp = classify(&fields).unwrap();
197        assert_eq!(mcp.category, McpCategory::Lifecycle);
198        assert_eq!(mcp.msg_type, McpMsgType::Notification);
199    }

203        let body = br#"{"jsonrpc":"2.0","result":{"tools":[]},"id":1}"#;

207    }
