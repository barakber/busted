4/// SDK/agent info extracted from User-Agent header.
5#[derive(Debug, Clone)]
6pub struct SdkInfo {
7    /// SDK or library name (e.g. `"openai-python"`, `"anthropic-typescript"`).
8    pub name: String,
9    /// Version string if parseable from the User-Agent.
10    pub version: Option<String>,
11}

13/// Model parameters extracted from the request body.
14#[derive(Debug, Clone, Default)]
15pub struct ModelParams {
16    /// Model identifier (e.g. `"gpt-4"`, `"claude-3-opus"`).
17    pub model: Option<String>,
18    /// Sampling temperature.
19    pub temperature: Option<f64>,
20    /// Maximum tokens to generate.
21    pub max_tokens: Option<u64>,

26/// Complete agent fingerprint.
27#[derive(Debug, Clone)]
28pub struct AgentFingerprint {
29    /// SDK/library detected from User-Agent.
30    pub sdk: Option<SdkInfo>,
31    /// API version header (e.g. Anthropic's `anthropic-version`).
32    pub api_version: Option<String>,
33    /// Model parameters extracted from the request body.
34    pub model_params: ModelParams,
35    /// FNV-1a hash of request structure for behavioral grouping.
36    pub signature_hash: u64,

39/// Known SDK patterns in User-Agent strings: (substring, sdk_name).
40const SDK_PATTERNS: &[(&str, &str)] = &[

42    ("openai-node", "openai-node"),
43    ("openai-java", "openai-java"),

58/// Detect SDK from a User-Agent string.
59pub fn detect_sdk(user_agent: &str) -> Option<SdkInfo> {
60    let ua_lower = user_agent.to_lowercase();
61
62    for &(pattern, name) in SDK_PATTERNS {
63        if ua_lower.contains(pattern) {
64            let version = extract_version(user_agent, pattern);
65            return Some(SdkInfo {
66                name: name.to_string(),
67                version,
68            });
69        }
70    }
71
72    None
73}

75/// Extract version string following an SDK name pattern.
76/// Looks for patterns like "sdk-name/1.2.3" or "sdk-name 1.2.3".
77fn extract_version(ua: &str, pattern: &str) -> Option<String> {
78    let lower = ua.to_lowercase();
79    let idx = lower.find(pattern)?;
80    let after = &ua[idx + pattern.len()..];
81
82    // Skip separator (/ or space)
83    let after = after
84        .strip_prefix('/')
85        .or_else(|| after.strip_prefix(' '))?;
86
87    // Take version chars (digits, dots, dashes, alphanumeric)
88    let version: String = after
89        .chars()
90        .take_while(|c| c.is_ascii_alphanumeric() || *c == '.' || *c == '-')
91        .collect();
92
93    if version.is_empty() {
94        None
95    } else {
96        Some(version)
97    }
98}

100/// Extract API version from provider-specific headers.
101pub fn extract_api_version(headers: &std::collections::HashMap<String, String>) -> Option<String> {
102    // Anthropic
103    if let Some(v) = headers.get("anthropic-version") {
104        return Some(v.clone());
105    }
106    // OpenAI doesn't have a version header but has organization
107    // Azure has api-version query param (not in headers usually)
108    None
109}

111/// Compute FNV-1a hash of the request "shape" for fingerprinting.
112///
113/// Hashes: method + path + sorted header names + ordered JSON body keys.
114/// Different SDKs produce different key orderings and header sets.
115pub fn compute_signature_hash(req: &HttpRequestInfo, json: &JsonFields) -> u64 {
116    let mut hasher = Fnv1a::new();
117
118    // Method + path
119    hasher.write(req.method.as_bytes());
120    hasher.write(req.path.as_bytes());
121
122    // Sorted header names (not values — values change per request)
123    let mut header_names: Vec<&str> = req.headers.keys().map(|k| k.as_str()).collect();
124    header_names.sort();
125    for name in header_names {
126        hasher.write(name.as_bytes());
127    }
128
129    // Ordered JSON body keys (order matters — it's SDK-specific)
130    for key in &json.top_level_keys {
131        hasher.write(key.as_bytes());
132    }
133
134    hasher.finish()
135}

137/// Build a complete fingerprint from HTTP request and JSON body.
138pub fn build_fingerprint(req: &HttpRequestInfo, json: &JsonFields) -> AgentFingerprint {
139    let sdk = req.headers.get("user-agent").and_then(|ua| detect_sdk(ua));
140
141    let api_version = extract_api_version(&req.headers);
142
143    let model_params = ModelParams {
144        model: json.model.clone(),
145        temperature: json.temperature,
146        max_tokens: json.max_tokens,
147        stream: json.stream,
148    };
149
150    let signature_hash = compute_signature_hash(req, json);
151
152    AgentFingerprint {
153        sdk,
154        api_version,
155        model_params,
156        signature_hash,
157    }
158}

160/// FNV-1a 64-bit hasher.
161struct Fnv1a {
162    state: u64,
163}
164
165impl Fnv1a {
166    const OFFSET_BASIS: u64 = 0xcbf29ce484222325;
167    const PRIME: u64 = 0x00000100000001B3;
168
169    fn new() -> Self {
170        Fnv1a {
171            state: Self::OFFSET_BASIS,
172        }
173    }
174
175    fn write(&mut self, bytes: &[u8]) {
176        for &byte in bytes {
177            self.state ^= byte as u64;
178            self.state = self.state.wrapping_mul(Self::PRIME);
179        }
180    }
181
182    fn finish(&self) -> u64 {
183        self.state
184    }
185}

224    #[test]
225    fn test_signature_hash_stability() {
226        use crate::http;
227        use crate::json;
228
229        let raw = b"POST /v1/chat/completions HTTP/1.1\r\nHost: api.openai.com\r\nUser-Agent: openai-python/1.12.0\r\n\r\n{\"model\":\"gpt-4\",\"messages\":[]}";
230        let req = http::parse_request(raw).unwrap();
231        let body = &raw[req.body_offset.unwrap()..];
232        let jf = json::analyze(body);
233
234        let h1 = compute_signature_hash(&req, &jf);
235        let h2 = compute_signature_hash(&req, &jf);
236        assert_eq!(h1, h2); // Same input → same hash
237
238        // Different path → different hash
239        let raw2 = b"POST /v1/embeddings HTTP/1.1\r\nHost: api.openai.com\r\nUser-Agent: openai-python/1.12.0\r\n\r\n{\"model\":\"gpt-4\",\"messages\":[]}";
240        let req2 = http::parse_request(raw2).unwrap();
241        let h3 = compute_signature_hash(&req2, &jf);
242        assert_ne!(h1, h3);
243    }
