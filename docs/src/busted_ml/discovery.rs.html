1/// Unsupervised pattern discovery via HDBSCAN clustering.
2/// Buffers feature vectors and periodically reclusters to discover
3/// novel traffic patterns not captured by the supervised classifier.
4pub struct PatternDiscovery {
5    feature_buffer: Vec<Vec<f32>>,
6    cluster_labels: Vec<i32>,
7    max_buffer_size: usize,
8    min_cluster_size: usize,
9}
10
11impl PatternDiscovery {
12    pub fn new() -> Self {
13        PatternDiscovery {
14            feature_buffer: Vec::new(),
15            cluster_labels: Vec::new(),
16            max_buffer_size: 2000,
17            min_cluster_size: 10,
18        }
19    }
20
21    /// Ingest a feature vector and return its cluster assignment.
22    /// Returns -1 (noise) until enough data accumulates for clustering.
23    pub fn ingest(&mut self, features: &[f64]) -> i32 {
24        // Convert f64 features to f32 for hdbscan
25        let f32_features: Vec<f32> = features.iter().map(|&v| v as f32).collect();
26        self.feature_buffer.push(f32_features);
27
28        let idx = self.feature_buffer.len() - 1;
29
30        // Recluster when buffer is full
31        if self.feature_buffer.len() >= self.max_buffer_size {
32            self.recluster();
33        }
34
35        // Return label if available
36        if idx < self.cluster_labels.len() {
37            self.cluster_labels[idx]
38        } else {
39            -1 // Not yet clustered
40        }
41    }
42
43    /// Run HDBSCAN on the buffered features.
44    fn recluster(&mut self) {
45        if self.feature_buffer.len() < self.min_cluster_size * 2 {
46            return;
47        }
48
49        let hyper_params = hdbscan::HdbscanHyperParams::builder()
50            .min_cluster_size(self.min_cluster_size)
51            .build();
52
53        let clusterer = hdbscan::Hdbscan::new(&self.feature_buffer, hyper_params);
54        match clusterer.cluster() {
55            Ok(labels) => {
56                let n_clusters = labels
57                    .iter()
58                    .filter(|&&l| l >= 0)
59                    .max()
60                    .map_or(0, |m| m + 1);
61                log::info!(
62                    "HDBSCAN: {} clusters found in {} samples ({} noise)",
63                    n_clusters,
64                    labels.len(),
65                    labels.iter().filter(|&&l| l < 0).count()
66                );
67                self.cluster_labels = labels;
68            }
69            Err(e) => {
70                log::warn!("HDBSCAN clustering failed: {:?}", e);
71            }
72        }
73
74        // FIFO eviction: keep the most recent half
75        if self.feature_buffer.len() >= self.max_buffer_size {
76            let half = self.feature_buffer.len() / 2;
77            self.feature_buffer.drain(0..half);
78            self.cluster_labels.clear(); // invalidate labels until next recluster
79        }
80    }
81
82    #[allow(dead_code)]
83    pub fn last_cluster_label(&self) -> i32 {
84        self.cluster_labels.last().copied().unwrap_or(-1)
85    }
86
87    #[allow(dead_code)]
88    pub fn num_clusters(&self) -> usize {
89        if self.cluster_labels.is_empty() {
90            return 0;
91        }
92        self.cluster_labels
93            .iter()
94            .filter(|&&l| l >= 0)
95            .max()
96            .map_or(0, |&m| (m + 1) as usize)
97    }
98}

100#[cfg(test)]
101mod tests {
102    use super::*;
103    use crate::features::FEATURE_DIM;
104
105    #[test]
106    fn new_num_clusters_zero() {
107        let d = PatternDiscovery::new();
108        assert_eq!(d.num_clusters(), 0);
109    }
110
111    #[test]
112    fn new_last_cluster_label_negative() {
113        let d = PatternDiscovery::new();
114        assert_eq!(d.last_cluster_label(), -1);
115    }
116
117    #[test]
118    fn ingest_returns_negative_initially() {
119        let mut d = PatternDiscovery::new();
120        let features = vec![0.0f64; FEATURE_DIM];
121        let label = d.ingest(&features);
122        assert_eq!(label, -1);
123    }
124
125    #[test]
126    fn buffer_under_limit_no_panic() {
127        let mut d = PatternDiscovery::new();
128        for i in 0..100 {
129            let features: Vec<f64> = (0..FEATURE_DIM).map(|j| (i * j) as f64 / 1000.0).collect();
130            d.ingest(&features);
131        }
132        assert_eq!(d.feature_buffer.len(), 100);
133    }
134
135    #[test]
136    fn buffer_at_max_triggers_recluster_no_panic() {
137        let mut d = PatternDiscovery::new();
138        // Fill to max_buffer_size (2000)
139        for i in 0..2000 {
140            let features: Vec<f64> = (0..FEATURE_DIM)
141                .map(|j| ((i + j) % 100) as f64 / 100.0)
142                .collect();
143            d.ingest(&features);
144        }
145        // After recluster + FIFO eviction, buffer should be smaller
146        assert!(d.feature_buffer.len() < 2000);
147    }
148}
