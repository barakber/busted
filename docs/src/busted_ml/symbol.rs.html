1use busted_types::NetworkEvent;
2
3/// Compact symbolic representation of a network event.
4/// 6 kinds x 6 sizes x 5 ports = 180 possible symbols.
5#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
6pub struct Symbol {
7    /// Connect=0, Send=1, Recv=2, Close=3, Dns=4, Unknown=5
8    pub kind: u8,
9    /// 0=zero, 1=tiny(<128), 2=small(<1K), 3=med(<8K), 4=large(<64K), 5=huge
10    pub size_bucket: u8,
11    /// 0=443, 1=80, 2=53, 3=other-well-known(<1024), 4=ephemeral
12    pub port_class: u8,
13}
14
15const NUM_KINDS: u16 = 6;
16const NUM_SIZES: u16 = 6;
17const NUM_PORTS: u16 = 5;
18
19/// Total number of distinct symbols.
20pub const SYMBOL_SPACE: usize = (NUM_KINDS * NUM_SIZES * NUM_PORTS) as usize; // 180
21
22impl Symbol {
23    pub fn from_network_event(event: &NetworkEvent) -> Self {
24        let kind = match event.event_type {
25            1 => 0, // TCP_CONNECT
26            2 => 1, // DATA_SENT
27            3 => 2, // DATA_RECEIVED
28            4 => 3, // CONNECTION_CLOSED
29            5 => 4, // DNS_QUERY
30            _ => 5, // Unknown
31        };
32
33        let size_bucket = match event.bytes {
34            0 => 0,
35            1..=127 => 1,
36            128..=1023 => 2,
37            1024..=8191 => 3,
38            8192..=65535 => 4,
39            _ => 5,
40        };
41
42        let port_class = match event.dport {
43            443 => 0,
44            80 => 1,
45            53 => 2,
46            0..=1023 => 3,
47            _ => 4,
48        };
49
50        Symbol {
51            kind,
52            size_bucket,
53            port_class,
54        }
55    }
56
57    /// Encode symbol as a unique index in [0, 180).
58    pub fn encode(&self) -> u16 {
59        self.kind as u16 * NUM_SIZES * NUM_PORTS
60            + self.size_bucket as u16 * NUM_PORTS
61            + self.port_class as u16
62    }
63}
