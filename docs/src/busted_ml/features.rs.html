8/// Total feature vector dimensionality.
9///
10/// | Range     | Count | Feature Group                                          |
11/// |-----------|-------|--------------------------------------------------------|
12/// | 0..180    |  180  | Unigram histogram (normalized symbol frequencies)      |
13/// | 180..280  |  100  | Bigram histogram (hashed consecutive pairs)             |
14/// | 280..330  |   50  | Trigram histogram (hashed triples)                      |
15/// | 330..336  |    6  | Timing: inter-arrival mean/std/min/max/p50/p90          |
16/// | 336..342  |    6  | Bytes: send/recv ratio, mean send/recv, var, zero frac  |
17/// | 342..346  |    4  | Connection: IP diversity, port div, unique dst, reuse   |
18/// | 346..350  |    4  | Burst: count, mean len, max gap, burst byte fraction    |
19/// | 350..352  |    2  | Entropy: symbol entropy, bigram entropy                 |
20pub const FEATURE_DIM: usize = 352;
21
22const BIGRAM_BINS: usize = 100;
23const TRIGRAM_BINS: usize = 50;
24
25pub struct FeatureExtractor {
26    scratch: Vec<f64>,
27}
28
29impl FeatureExtractor {
30    pub fn new() -> Self {
31        FeatureExtractor {
32            scratch: vec![0.0; FEATURE_DIM],
33        }
34    }
35
36    pub fn extract(&mut self, window: &EventWindow) -> Array1<f64> {
37        self.scratch.fill(0.0);
38
39        let syms: Vec<_> = window.symbols().collect();
40        let n = syms.len() as f64;
41        if n < 1.0 {
42            return Array1::zeros(FEATURE_DIM);
43        }
44
45        // ---- Unigrams (0..180) ----
46        for ts in &syms {
47            let idx = ts.symbol.encode() as usize;
48            if idx < SYMBOL_SPACE {
49                self.scratch[idx] += 1.0 / n;
50            }
51        }
52
53        // ---- Bigrams (180..280) ----
54        for pair in syms.windows(2) {
55            let a = pair[0].symbol.encode() as u64;
56            let b = pair[1].symbol.encode() as u64;
57            let hash = fnv1a_u64(a.wrapping_mul(181).wrapping_add(b));
58            let bin = (hash as usize) % BIGRAM_BINS;
59            self.scratch[SYMBOL_SPACE + bin] += 1.0 / (n - 1.0).max(1.0);
60        }
61
62        // ---- Trigrams (280..330) ----
63        for triple in syms.windows(3) {
64            let a = triple[0].symbol.encode() as u64;
65            let b = triple[1].symbol.encode() as u64;
66            let c = triple[2].symbol.encode() as u64;
67            let hash = fnv1a_u64(
68                a.wrapping_mul(32761)
69                    .wrapping_add(b.wrapping_mul(181))
70                    .wrapping_add(c),
71            );
72            let bin = (hash as usize) % TRIGRAM_BINS;
73            self.scratch[SYMBOL_SPACE + BIGRAM_BINS + bin] += 1.0 / (n - 2.0).max(1.0);
74        }
75
76        // ---- Timing features (330..336) ----
77        let timing_offset = SYMBOL_SPACE + BIGRAM_BINS + TRIGRAM_BINS;
78        let mut deltas = Vec::new();
79        for pair in syms.windows(2) {
80            let dt = pair[1].timestamp_ns.saturating_sub(pair[0].timestamp_ns) as f64;
81            deltas.push(dt);
82        }
83        if !deltas.is_empty() {
84            deltas.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
85            let mean = deltas.iter().sum::<f64>() / deltas.len() as f64;
86            let variance =
87                deltas.iter().map(|d| (d - mean).powi(2)).sum::<f64>() / deltas.len() as f64;
88            let std_dev = variance.sqrt();
89            let min = deltas[0];
90            let max = deltas[deltas.len() - 1];
91            let p50 = percentile(&deltas, 0.5);
92            let p90 = percentile(&deltas, 0.9);
93
94            // Normalize to milliseconds for more reasonable feature scales
95            self.scratch[timing_offset] = mean / 1_000_000.0;
96            self.scratch[timing_offset + 1] = std_dev / 1_000_000.0;
97            self.scratch[timing_offset + 2] = min / 1_000_000.0;
98            self.scratch[timing_offset + 3] = max / 1_000_000.0;
99            self.scratch[timing_offset + 4] = p50 / 1_000_000.0;
100            self.scratch[timing_offset + 5] = p90 / 1_000_000.0;
101        }
102
103        // ---- Bytes features (336..342) ----
104        let bytes_offset = timing_offset + 6;
105        let mut send_bytes = Vec::new();
106        let mut recv_bytes = Vec::new();
107        let mut zero_count = 0u64;
108        for ts in &syms {
109            match ts.symbol.kind {
110                1 => send_bytes.push(ts.bytes as f64), // DATA_SENT
111                2 => recv_bytes.push(ts.bytes as f64), // DATA_RECEIVED
112                _ => {}
113            }
114            if ts.bytes == 0 {
115                zero_count += 1;
116            }
117        }
118        let total_send: f64 = send_bytes.iter().sum();
119        let total_recv: f64 = recv_bytes.iter().sum();
120        // Send/recv ratio
121        self.scratch[bytes_offset] = if total_recv > 0.0 {
122            total_send / total_recv
123        } else if total_send > 0.0 {
124            10.0 // cap
125        } else {
126            0.0
127        };
128        // Mean send size
129        self.scratch[bytes_offset + 1] = if !send_bytes.is_empty() {
130            total_send / send_bytes.len() as f64 / 1024.0
131        } else {
132            0.0
133        };
134        // Mean recv size
135        self.scratch[bytes_offset + 2] = if !recv_bytes.is_empty() {
136            total_recv / recv_bytes.len() as f64 / 1024.0
137        } else {
138            0.0
139        };
140        // Byte variance (all events)
141        let all_bytes: Vec<f64> = syms.iter().map(|ts| ts.bytes as f64).collect();
142        let byte_mean = all_bytes.iter().sum::<f64>() / n;
143        self.scratch[bytes_offset + 3] = all_bytes
144            .iter()
145            .map(|b| (b - byte_mean).powi(2))
146            .sum::<f64>()
147            / n
148            / 1_000_000.0; // normalize
149                           // Zero-byte fraction
150        self.scratch[bytes_offset + 4] = zero_count as f64 / n;
151        // Total bytes (normalized to KB)
152        self.scratch[bytes_offset + 5] = (total_send + total_recv) / 1024.0;
153
154        // ---- Connection features (342..346) ----
155        let conn_offset = bytes_offset + 6;
156        let unique_ips: HashSet<u32> = syms.iter().map(|ts| ts.dst_ip_hash).collect();
157        let unique_ports: HashSet<u16> = syms.iter().map(|ts| ts.dst_port).collect();
158        self.scratch[conn_offset] = unique_ips.len() as f64 / n.max(1.0); // IP diversity
159        self.scratch[conn_offset + 1] = unique_ports.len() as f64 / n.max(1.0); // Port diversity
160        self.scratch[conn_offset + 2] = unique_ips.len() as f64; // Unique dst count
161                                                                 // Connection reuse ratio: events / unique destinations
162        self.scratch[conn_offset + 3] = if unique_ips.is_empty() {
163            0.0
164        } else {
165            n / unique_ips.len() as f64
166        };
167
168        // ---- Burst features (346..350) ----
169        let burst_offset = conn_offset + 4;
170        let burst_gap_ms = 500.0; // events within 500ms are a burst
171        let mut bursts: Vec<(usize, u64)> = Vec::new(); // (length, bytes)
172        let mut burst_len = 1usize;
173        let mut burst_bytes = syms.first().map_or(0, |ts| ts.bytes);
174        let mut max_gap_ms: f64 = 0.0;
175        for pair in syms.windows(2) {
176            let dt_ms =
177                pair[1].timestamp_ns.saturating_sub(pair[0].timestamp_ns) as f64 / 1_000_000.0;
178            max_gap_ms = max_gap_ms.max(dt_ms);
179            if dt_ms < burst_gap_ms {
180                burst_len += 1;
181                burst_bytes += pair[1].bytes;
182            } else {
183                if burst_len > 1 {
184                    bursts.push((burst_len, burst_bytes));
185                }
186                burst_len = 1;
187                burst_bytes = pair[1].bytes;
188            }
189        }
190        if burst_len > 1 {
191            bursts.push((burst_len, burst_bytes));
192        }
193        self.scratch[burst_offset] = bursts.len() as f64; // burst count
194        self.scratch[burst_offset + 1] = if bursts.is_empty() {
195            0.0
196        } else {
197            bursts.iter().map(|(l, _)| *l as f64).sum::<f64>() / bursts.len() as f64
198        }; // mean burst length
199        self.scratch[burst_offset + 2] = max_gap_ms / 1000.0; // max gap in seconds
200        let total_burst_bytes: u64 = bursts.iter().map(|(_, b)| b).sum();
201        let total_all_bytes: u64 = syms.iter().map(|ts| ts.bytes).sum();
202        self.scratch[burst_offset + 3] = if total_all_bytes > 0 {
203            total_burst_bytes as f64 / total_all_bytes as f64
204        } else {
205            0.0
206        }; // burst byte fraction
207
208        // ---- Entropy features (350..352) ----
209        let entropy_offset = burst_offset + 4;
210        // Symbol entropy
211        self.scratch[entropy_offset] = shannon_entropy(&self.scratch[0..SYMBOL_SPACE]);
212        // Bigram entropy
213        self.scratch[entropy_offset + 1] =
214            shannon_entropy(&self.scratch[SYMBOL_SPACE..SYMBOL_SPACE + BIGRAM_BINS]);
215
216        Array1::from_vec(self.scratch.clone())
217    }
218}

220fn percentile(sorted: &[f64], p: f64) -> f64 {
221    if sorted.is_empty() {
222        return 0.0;
223    }
224    let idx = (p * (sorted.len() - 1) as f64) as usize;
225    sorted[idx.min(sorted.len() - 1)]
226}
227
228fn shannon_entropy(probs: &[f64]) -> f64 {
229    let mut ent = 0.0;
230    for &p in probs {
231        if p > 0.0 {
232            ent -= p * p.log2();
233        }
234    }
235    ent
236}

238/// Simple FNV-1a hash for u64 values.
239fn fnv1a_u64(val: u64) -> u64 {
240    let mut h: u64 = 0xcbf29ce484222325;
241    for &byte in &val.to_le_bytes() {
242        h ^= byte as u64;
243        h = h.wrapping_mul(0x100000001b3);
244    }
245    h
