2//!
3//! All colors are defined as `(u8, u8, u8)` RGB tuples so that
4//! framework-specific crates (`ratatui`, `egui`) can convert them
5//! to their own color types without pulling in either dependency.

47/// Map an action variant to its theme RGB color.
48pub fn action_color(action: &AgenticAction) -> Rgb {
49    match action {
50        AgenticAction::Prompt { .. } => PROMPT,
51        AgenticAction::Response { .. } => RESPONSE,
52        AgenticAction::ToolCall { .. } | AgenticAction::ToolResult { .. } => TOOL,
53        AgenticAction::McpRequest { .. } | AgenticAction::McpResponse { .. } => MCP,
54        AgenticAction::PiiDetected { .. } => PII,
55        AgenticAction::Network { .. } => NETWORK,
56    }
57}
58
59/// Map a policy decision string to its theme RGB color.
60pub fn policy_color(policy: Option<&str>) -> Rgb {
61    match policy {
62        Some("allow") => ALLOW,
63        Some("audit") => AUDIT,
64        Some("deny") => DENY,
65        _ => DIM_TEXT,
66    }
67}

69/// Tint a color toward a background: blend `ratio` of `bg` into `fg`.
70/// `ratio` 0.0 = pure fg, 1.0 = pure bg.
71pub fn tint(fg: Rgb, bg: Rgb, ratio: f32) -> Rgb {
72    let r = ratio.clamp(0.0, 1.0);
73    let mix = |a: u8, b: u8| -> u8 { ((a as f32 * (1.0 - r)) + (b as f32 * r)) as u8 };
74    (mix(fg.0, bg.0), mix(fg.1, bg.1), mix(fg.2, bg.2))
75}
76
77/// Create a subtle background tint for pill badges: mostly `bg` with a hint of `fg`.
78pub fn pill_bg(fg: Rgb, bg: Rgb) -> Rgb {
79    tint(fg, bg, 0.85)
