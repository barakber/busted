9const SOCKET_PATH: &str = "/tmp/busted.sock";
10
11/// Configuration for the UI.
12pub struct UiConfig {
13    pub demo_mode: bool,
14}
15
16/// Launch the egui dashboard.
17pub fn run_ui(config: UiConfig) -> eframe::Result<()> {
18    let (tx, rx) = mpsc::channel();
19
20    if config.demo_mode {
21        demo::start(tx);
22    } else {
23        // Background thread: connect to agent's Unix socket and forward events
24        std::thread::spawn(move || {
25            loop {
26                match std::os::unix::net::UnixStream::connect(SOCKET_PATH) {
27                    Ok(stream) => {
28                        let reader = std::io::BufReader::new(stream);
29                        for line in reader.lines() {
30                            match line {
31                                Ok(line) if !line.is_empty() => {
32                                    if let Ok(event) = serde_json::from_str(&line) {
33                                        if tx.send(event).is_err() {
34                                            return; // UI closed
35                                        }
36                                    }
37                                }
38                                Err(_) => break, // Connection lost, retry
39                                _ => {}
40                            }
41                        }
42                    }
43                    Err(_) => {
44                        // Agent not running, retry after delay
45                    }
46                }
47                std::thread::sleep(std::time::Duration::from_secs(2));
48            }
49        });
50    }
51
52    let options = eframe::NativeOptions {
53        viewport: egui::ViewportBuilder::default()
54            .with_inner_size([1100.0, 700.0])
55            .with_title("Busted - LLM Communication Monitor"),
56        ..Default::default()
57    };
58
59    eframe::run_native(
60        "Busted",
61        options,
62        Box::new(move |_cc| Ok(Box::new(BustedApp::new(rx, config.demo_mode)))),
63    )
64}
