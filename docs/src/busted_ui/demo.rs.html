117fn now_timestamp() -> String {
118    let dur = SystemTime::now()
119        .duration_since(UNIX_EPOCH)
120        .unwrap_or_default();
121    let total_secs = dur.as_secs();
122    let hours = (total_secs / 3600) % 24;
123    let minutes = (total_secs / 60) % 60;
124    let seconds = total_secs % 60;
125    let millis = dur.subsec_millis();
126    format!("{:02}:{:02}:{:02}.{:03}", hours, minutes, seconds, millis)
127}

129fn make_event(scenario: &Scenario, event_type: &str, bytes: u64) -> ProcessedEvent {
130    let is_tls = event_type.starts_with("TLS_DATA_");
131    ProcessedEvent {
132        event_type: event_type.to_string(),
133        timestamp: now_timestamp(),
134        pid: scenario.pid,
135        uid: 1000,
136        process_name: scenario.process_name.to_string(),
137        src_ip: "192.168.1.100".to_string(),
138        src_port: 40000 + (scenario.pid as u16 % 10000),
139        dst_ip: scenario.dst_ip.to_string(),
140        dst_port: scenario.dst_port,
141        bytes,
142        provider: Some(scenario.provider.to_string()),
143        policy: Some("audit".to_string()),
144        container_id: String::new(),
145        cgroup_id: 0,
146        request_rate: Some(2.5),
147        session_bytes: Some(bytes),
148        pod_name: None,
149        pod_namespace: None,
150        service_account: None,
151        ml_confidence: None,
152        ml_provider: None,
153        behavior_class: None,
154        cluster_id: None,
155        sni: Some(scenario.sni.to_string()),
156        tls_protocol: if is_tls {
157            Some("LLM_API".to_string())
158        } else {
159            None
160        },
161        tls_details: if is_tls {
162            scenario
163                .model
164                .map(|m| format!("{} model={}", scenario.provider, m))
165        } else {
166            None
167        },
168        tls_payload: None,
169        content_class: if is_tls {
170            Some("LLM_API".to_string())
171        } else {
172            None
173        },
174        llm_provider: Some(scenario.provider.to_string()),
175        llm_endpoint: if is_tls {
176            Some("/v1/chat/completions".to_string())
177        } else {
178            None
179        },
180        llm_model: scenario.model.map(|s| s.to_string()),
181        mcp_method: scenario.mcp_method.map(|s| s.to_string()),
182        mcp_category: scenario.mcp_category.map(|s| s.to_string()),
183        agent_sdk: scenario.sdk.map(|s| s.to_string()),
184        agent_fingerprint: Some(0xdeadbeef),
185        classifier_confidence: if is_tls { Some(0.95) } else { None },
186        pii_detected: if is_tls { Some(false) } else { None },
187        llm_user_message: None,
188        llm_system_prompt: None,
189        llm_messages_json: None,
190        llm_stream: None,
191    }
192}

194pub fn start(tx: mpsc::Sender<ProcessedEvent>) {
195    std::thread::spawn(move || {
196        let mut cycle = 0usize;
197        loop {
198            let scenario = &SCENARIOS[cycle % SCENARIOS.len()];
199
200            // TCP_CONNECT
201            let event = make_event(scenario, "TCP_CONNECT", 0);
202            if tx.send(event).is_err() {
203                return;
204            }
205            std::thread::sleep(Duration::from_millis(50));
206
207            // TLS_DATA_WRITE (request)
208            let request_bytes = 256 + (cycle as u64 * 37) % 512;
209            let event = make_event(scenario, "TLS_DATA_WRITE", request_bytes);
210            if tx.send(event).is_err() {
211                return;
212            }
213            std::thread::sleep(Duration::from_millis(50));
214
215            // TLS_DATA_READ (response)
216            let response_bytes = 1024 + (cycle as u64 * 73) % 4096;
217            let event = make_event(scenario, "TLS_DATA_READ", response_bytes);
218            if tx.send(event).is_err() {
219                return;
220            }
221
222            cycle += 1;
223
224            // 300-500ms between scenarios
225            let delay = 300 + (cycle * 29) % 200;
226            std::thread::sleep(Duration::from_millis(delay as u64));
227        }
228    });
