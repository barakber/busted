59/// Draw a rounded pill badge. Returns the Response for interaction.
60pub fn pill(ui: &mut egui::Ui, text: &str, fg: Color32, bg: Color32) -> egui::Response {
61    let galley =
62        ui.fonts_mut(|f| f.layout_no_wrap(text.to_string(), egui::FontId::proportional(11.0), fg));
63    let text_size = galley.size();
64    let padding = egui::vec2(6.0, 2.0);
65    let desired = text_size + padding * 2.0;
66    let (rect, response) = ui.allocate_exact_size(desired, egui::Sense::hover());
67    if ui.is_rect_visible(rect) {
68        ui.painter()
69            .rect_filled(rect, CornerRadius::same(ROUNDING), bg);
70        ui.painter().galley(rect.min + padding, galley, fg);
71    }
72    response
73}
