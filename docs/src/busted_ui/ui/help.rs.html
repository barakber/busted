37pub fn draw(ctx: &egui::Context) {
38    egui::Window::new("Help")
39        .collapsible(false)
40        .resizable(false)
41        .anchor(egui::Align2::CENTER_CENTER, [0.0, 0.0])
42        .frame(style::card_frame().fill(style::SURFACE))
43        .show(ctx, |ui| {
44            ui.colored_label(
45                style::BRIGHT_TEXT,
46                egui::RichText::new("Key Bindings").strong().size(16.0),
47            );
48            ui.add_space(8.0);
49
50            for (key, desc) in BINDINGS {
51                if key.is_empty() {
52                    ui.add_space(6.0);
53                    continue;
54                }
55                if desc.is_empty() {
56                    // Section header
57                    ui.colored_label(
58                        style::ACCENT_COLOR,
59                        egui::RichText::new(*key).strong().size(12.0),
60                    );
61                    // Separator
62                    let rect = ui.max_rect();
63                    let y = ui.cursor().top();
64                    ui.painter().line_segment(
65                        [egui::pos2(rect.left(), y), egui::pos2(rect.right(), y)],
66                        egui::Stroke::new(1.0, style::BORDER_COLOR),
67                    );
68                    ui.add_space(2.0);
69                    continue;
70                }
71                ui.horizontal(|ui| {
72                    // Key as pill
73                    style::pill(ui, key, style::ACCENT_COLOR, style::ACCENT_DIM);
74                    ui.add_space(8.0);
75                    ui.colored_label(style::NORMAL_TEXT, *desc);
76                });
77            }
78
79            ui.add_space(8.0);
80            ui.colored_label(style::DIM_TEXT, "Press ? or Esc to close");
81        });
82}
