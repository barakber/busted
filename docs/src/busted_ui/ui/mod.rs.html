25/// The main eframe application that holds the App state and event bridge.
26pub struct BustedApp {
27    pub app: App,
28    rx: Option<mpsc::Receiver<BustedEvent>>,
29    last_tick: Instant,
30    /// For WASM demo mode: cycle counter and per-event timing
31    #[cfg(target_arch = "wasm32")]
32    wasm_cycle: usize,
33    #[cfg(target_arch = "wasm32")]
34    wasm_batch: Vec<(BustedEvent, u64)>,
35    #[cfg(target_arch = "wasm32")]
36    wasm_batch_idx: usize,
37    #[cfg(target_arch = "wasm32")]
38    wasm_last_event: f64,
39}
40
41impl BustedApp {
42    /// Create a new BustedApp. On native, pass the mpsc receiver from bridge.
43    #[cfg(not(target_arch = "wasm32"))]
44    pub fn new(app: App, rx: mpsc::Receiver<BustedEvent>) -> Self {
45        Self {
46            app,
47            rx: Some(rx),
48            last_tick: Instant::now(),
49        }
50    }
51
52    /// Create a new BustedApp for WASM (demo-only, no receiver).
53    #[cfg(target_arch = "wasm32")]
54    pub fn new_wasm(app: App) -> Self {
55        Self {
56            app,
57            rx: None,
58            last_tick: Instant::now(),
59            wasm_cycle: 0,
60            wasm_batch: Vec::new(),
61            wasm_batch_idx: 0,
62            wasm_last_event: 0.0,
63        }
64    }
65
66    fn drain_events(&mut self) {
67        if let Some(ref rx) = self.rx {
68            while let Ok(event) = rx.try_recv() {
69                self.app.push_event(event);
70            }
71        }
72    }
73
74    fn tick_if_needed(&mut self) {
75        if self.last_tick.elapsed() >= Duration::from_secs(1) {
76            self.app.tick();
77            self.last_tick = Instant::now();
78        }
79    }
80
81    #[cfg(target_arch = "wasm32")]
82    fn wasm_demo_tick(&mut self, time: f64) {
83        if self.wasm_batch.is_empty() || self.wasm_batch_idx >= self.wasm_batch.len() {
84            self.wasm_batch = crate::demo::generate_batch(self.wasm_cycle);
85            self.wasm_batch_idx = 0;
86            self.wasm_cycle += 1;
87            self.wasm_last_event = time;
88        }
89
90        while self.wasm_batch_idx < self.wasm_batch.len() {
91            let (_, delay_ms) = &self.wasm_batch[self.wasm_batch_idx];
92            let elapsed_ms = (time - self.wasm_last_event) * 1000.0;
93            if elapsed_ms >= *delay_ms as f64 {
94                let (event, _) = self.wasm_batch[self.wasm_batch_idx].clone();
95                self.app.push_event(event);
96                self.wasm_last_event = time;
97                self.wasm_batch_idx += 1;
98            } else {
99                break;
100            }
101        }
102    }
103}

105impl eframe::App for BustedApp {
106    fn update(&mut self, ctx: &egui::Context, _frame: &mut eframe::Frame) {
107        // Drain events from bridge
108        self.drain_events();
109        self.tick_if_needed();
110
111        // WASM demo mode
112        #[cfg(target_arch = "wasm32")]
113        {
114            let time = ctx.input(|i| i.time);
115            self.wasm_demo_tick(time);
116            ctx.request_repaint_after(Duration::from_millis(100));
117        }
118
119        // Handle keyboard input
120        handle_keyboard(ctx, &mut self.app);
121
122        // Apply modern dark theme
123        let mut visuals = egui::Visuals::dark();
124        visuals.panel_fill = style::BG_COLOR;
125        visuals.window_fill = style::SURFACE;
126        visuals.override_text_color = Some(style::NORMAL_TEXT);
127        visuals.selection.bg_fill = style::ACCENT_DIM;
128        visuals.selection.stroke = egui::Stroke::new(1.0, style::ACCENT_COLOR);
129        // Rounded widgets
130        let rounding = egui::CornerRadius::same(style::ROUNDING);
131        visuals.widgets.noninteractive.corner_radius = rounding;
132        visuals.widgets.inactive.corner_radius = rounding;
133        visuals.widgets.hovered.corner_radius = rounding;
134        visuals.widgets.active.corner_radius = rounding;
135        visuals.widgets.noninteractive.bg_stroke = egui::Stroke::new(1.0, style::BORDER_COLOR);
136        ctx.set_visuals(visuals);
137
138        // Layout panels
139        egui::TopBottomPanel::top("header")
140            .exact_height(40.0)
141            .frame(
142                egui::Frame::NONE
143                    .fill(style::SURFACE)
144                    .inner_margin(egui::Margin::symmetric(16, 8)),
145            )
146            .show(ctx, |ui| {
147                header::draw(ui, &self.app);
148            });
149
150        egui::TopBottomPanel::top("stats")
151            .exact_height(56.0)
152            .frame(
153                egui::Frame::NONE
154                    .fill(style::BG_COLOR)
155                    .inner_margin(egui::Margin::symmetric(16, 4)),
156            )
157            .show(ctx, |ui| {
158                stats::draw(ui, &self.app);
159            });
160
161        egui::TopBottomPanel::top("tabs")
162            .exact_height(36.0)
163            .frame(
164                egui::Frame::NONE
165                    .fill(style::BG_COLOR)
166                    .inner_margin(egui::Margin::symmetric(16, 6)),
167            )
168            .show(ctx, |ui| {
169                tabs::draw(ui, &self.app);
170            });
171
172        egui::TopBottomPanel::top("filter")
173            .exact_height(36.0)
174            .frame(
175                egui::Frame::NONE
176                    .fill(style::BG_COLOR)
177                    .inner_margin(egui::Margin::symmetric(16, 4)),
178            )
179            .show(ctx, |ui| {
180                filter::draw(ui, &mut self.app);
181            });
182
183        egui::TopBottomPanel::bottom("footer")
184            .exact_height(28.0)
185            .frame(
186                egui::Frame::NONE
187                    .fill(style::SURFACE)
188                    .inner_margin(egui::Margin::symmetric(16, 4)),
189            )
190            .show(ctx, |ui| {
191                draw_footer(ui, &self.app);
192            });
193
194        // Detail side panel (right) â€” must be shown before CentralPanel
195        if self.app.show_detail && self.app.active_tab == Tab::Events {
196            egui::SidePanel::right("detail")
197                .default_width(400.0)
198                .min_width(300.0)
199                .frame(
200                    egui::Frame::NONE
201                        .fill(style::BG_COLOR)
202                        .inner_margin(egui::Margin::symmetric(12, 8))
203                        .stroke(egui::Stroke::new(2.0, style::ACCENT_COLOR)),
204                )
205                .show(ctx, |ui| {
206                    detail::draw(ui, &self.app);
207                });
208        }
209
210        // Content area
211        egui::CentralPanel::default()
212            .frame(
213                egui::Frame::NONE
214                    .fill(style::BG_COLOR)
215                    .inner_margin(egui::Margin::symmetric(16, 4)),
216            )
217            .show(ctx, |ui| {
218                self.app.content_height = ui.available_height();
219                draw_content(ui, &mut self.app);
220            });
221
222        // Help popup overlay
223        if self.app.show_help {
224            help::draw(ctx);
225        }
226    }
227}

229fn draw_content(ui: &mut egui::Ui, app: &mut App) {
230    match app.active_tab {
231        Tab::Events => events::draw(ui, app),
232        Tab::Processes => processes::draw(ui, app),
233        Tab::Providers => providers::draw(ui, app),
234    }
235}

237fn draw_footer(ui: &mut egui::Ui, app: &App) {
238    ui.horizontal(|ui| {
239        ui.spacing_mut().item_spacing.x = 4.0;
240
241        if app.show_detail {
242            footer_hint(ui, "Esc", "Close");
243            footer_hint(ui, "j/k", "Scroll");
244            footer_hint(ui, "C-d/u", "HalfPg");
245            footer_hint(ui, "n/N", "Next/Prev");
246            footer_hint(ui, "?", "Help");
247        } else if app.input_mode == InputMode::Search {
248            footer_hint(ui, "Enter", "Search");
249            footer_hint(ui, "Esc", "Cancel");
250        } else {
251            footer_hint(ui, "q", "Quit");
252            footer_hint(ui, "j/k", "Nav");
253            footer_hint(ui, "C-d/u", "HalfPg");
254            footer_hint(ui, "/", "Search");
255            footer_hint(ui, "Enter", "Detail");
256            footer_hint(ui, "Tab", "Tabs");
257            footer_hint(ui, "Space", "Pause");
258            footer_hint(ui, "?", "Help");
259        }
260    });
261}

269fn handle_keyboard(ctx: &egui::Context, app: &mut App) {
270    ctx.input(|input| {
271        let ctrl = input.modifiers.ctrl;
272
273        // Help popup intercepts all keys
274        if app.show_help {
275            if input.key_pressed(egui::Key::Escape) || input.key_pressed(egui::Key::Q) {
276                app.show_help = false;
277            }
278            // Check for '?' via events
279            for event in &input.events {
280                if let egui::Event::Text(t) = event {
281                    if t == "?" {
282                        app.show_help = false;
283                    }
284                }
285            }
286            return;
287        }
288
289        // Search mode
290        if app.input_mode == InputMode::Search {
291            if input.key_pressed(egui::Key::Escape) {
292                app.input_mode = InputMode::Normal;
293                return;
294            }
295            if input.key_pressed(egui::Key::Enter) {
296                app.input_mode = InputMode::Normal;
297                return;
298            }
299            if input.key_pressed(egui::Key::Backspace) {
300                app.search_query.pop();
301                return;
302            }
303            // Capture text input
304            for event in &input.events {
305                if let egui::Event::Text(t) = event {
306                    app.search_query.push_str(t);
307                }
308            }
309            return;
310        }
311
312        // Detail panel mode
313        if app.show_detail && app.active_tab == Tab::Events {
314            if input.key_pressed(egui::Key::Escape) {
315                app.show_detail = false;
316                return;
317            }
318            // Scroll detail content
319            if input.key_pressed(egui::Key::ArrowDown) {
320                app.detail_scroll_down();
321                return;
322            }
323            if input.key_pressed(egui::Key::ArrowUp) {
324                app.detail_scroll_up();
325                return;
326            }
327            if ctrl && input.key_pressed(egui::Key::D) {
328                app.detail_half_page_down();
329                return;
330            }
331            if ctrl && input.key_pressed(egui::Key::U) {
332                app.detail_half_page_up();
333                return;
334            }
335            if ctrl && input.key_pressed(egui::Key::F) {
336                app.detail_full_page_down();
337                return;
338            }
339            if ctrl && input.key_pressed(egui::Key::B) {
340                app.detail_full_page_up();
341                return;
342            }
343            if input.key_pressed(egui::Key::PageDown) {
344                app.detail_full_page_down();
345                return;
346            }
347            if input.key_pressed(egui::Key::PageUp) {
348                app.detail_full_page_up();
349                return;
350            }
351            if input.key_pressed(egui::Key::Home) {
352                app.detail_scroll = 0.0;
353                return;
354            }
355            if input.key_pressed(egui::Key::End) {
356                app.detail_scroll = f32::MAX / 2.0;
357                return;
358            }
359
360            // Text events for j/k/n/N/g/G/q/?
361            for event in &input.events {
362                if let egui::Event::Text(t) = event {
363                    match t.as_str() {
364                        "j" => {
365                            app.detail_scroll_down();
366                            return;
367                        }
368                        "k" => {
369                            app.detail_scroll_up();
370                            return;
371                        }
372                        "g" => {
373                            app.detail_scroll = 0.0;
374                            return;
375                        }
376                        "G" => {
377                            app.detail_scroll = f32::MAX / 2.0;
378                            return;
379                        }
380                        "n" => {
381                            app.select_next_event_in_detail();
382                            return;
383                        }
384                        "N" => {
385                            app.select_prev_event_in_detail();
386                            return;
387                        }
388                        "q" => { /* handled below as quit */ }
389                        "?" => {
390                            app.show_help = true;
391                            return;
392                        }
393                        _ => {}
394                    }
395                }
396            }
397            // Fall through for other keys
398        }
399
400        // Normal mode key presses
401        if ctrl && input.key_pressed(egui::Key::D) {
402            app.half_page_down();
403            return;
404        }
405        if ctrl && input.key_pressed(egui::Key::U) {
406            app.half_page_up();
407            return;
408        }
409        if ctrl && input.key_pressed(egui::Key::F) {
410            app.full_page_down();
411            return;
412        }
413        if ctrl && input.key_pressed(egui::Key::B) {
414            app.full_page_up();
415            return;
416        }
417        if input.key_pressed(egui::Key::ArrowDown) {
418            app.select_next();
419            return;
420        }
421        if input.key_pressed(egui::Key::ArrowUp) {
422            app.select_prev();
423            return;
424        }
425        if input.key_pressed(egui::Key::Home) {
426            app.select_first();
427            return;
428        }
429        if input.key_pressed(egui::Key::End) {
430            app.select_last();
431            return;
432        }
433        if input.key_pressed(egui::Key::PageDown) {
434            app.full_page_down();
435            return;
436        }
437        if input.key_pressed(egui::Key::PageUp) {
438            app.full_page_up();
439            return;
440        }
441        if input.key_pressed(egui::Key::Enter) {
442            app.toggle_detail();
443            return;
444        }
445        if input.key_pressed(egui::Key::Escape) {
446            if !app.search_query.is_empty() {
447                app.search_query.clear();
448            } else if app.show_detail {
449                app.show_detail = false;
450            }
451            return;
452        }
453        if input.key_pressed(egui::Key::Tab) {
454            if input.modifiers.shift {
455                app.active_tab = app.active_tab.prev();
456            } else {
457                app.active_tab = app.active_tab.next();
458            }
459            return;
460        }
461
462        // Handle text events for vim-style keys
463        for event in &input.events {
464            if let egui::Event::Text(t) = event {
465                match t.as_str() {
466                    "q" => {
467                        #[cfg(not(target_arch = "wasm32"))]
468                        std::process::exit(0);
469                    }
470                    "j" => {
471                        app.select_next();
472                        return;
473                    }
474                    "k" => {
475                        app.select_prev();
476                        return;
477                    }
478                    "g" => {
479                        app.select_first();
480                        return;
481                    }
482                    "G" => {
483                        app.select_last();
484                        return;
485                    }
486                    "H" => {
487                        app.select_high();
488                        return;
489                    }
490                    "M" => {
491                        app.select_mid();
492                        return;
493                    }
494                    "L" => {
495                        app.select_low();
496                        return;
497                    }
498                    " " => {
499                        app.toggle_pause();
500                        return;
501                    }
502                    "/" => {
503                        app.input_mode = InputMode::Search;
504                        return;
505                    }
506                    "1" => {
507                        app.active_tab = Tab::Events;
508                        return;
509                    }
510                    "2" => {
511                        app.active_tab = Tab::Processes;
512                        return;
513                    }
514                    "3" => {
515                        app.active_tab = Tab::Providers;
516                        return;
517                    }
518                    "c" => {
519                        app.clear_events();
520                        return;
521                    }
522                    "C" => {
523                        app.clear_filters();
524                        return;
525                    }
526                    "p" => {
527                        app.cycle_provider_filter();
528                        return;
529                    }
530                    "t" => {
531                        app.cycle_type_filter();
532                        return;
533                    }
534                    "?" => {
535                        app.show_help = true;
536                        return;
537                    }
538                    _ => {}
539                }
540            }
541        }
542    });
