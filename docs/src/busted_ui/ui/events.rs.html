7/// Pre-extracted row data to avoid borrow conflicts with the table builder.
8struct RowData {
9    time_short: String,
10    action_label: String,
11    action_color: egui::Color32,
12    process_name: String,
13    pid: String,
14    provider: String,
15    dest: String,
16    bytes_str: String,
17    policy: String,
18    policy_str: Option<String>,
19    is_pii: bool,
20}
21
22pub fn draw(ui: &mut egui::Ui, app: &mut App) {
23    let selected = app.selected_index;
24
25    // Pre-extract all row data while borrowing app immutably
26    let rows: Vec<RowData> = app
27        .filtered_events()
28        .iter()
29        .map(|ev| {
30            let action_color = style::action_color(&ev.action);
31            let arrow = action_arrow(&ev.action);
32            let action_type = ev.action_type();
33            let time = &ev.timestamp;
34            let time_short = if time.len() > 12 {
35                time[..12].to_string()
36            } else {
37                time.clone()
38            };
39            let bytes = ev.bytes();
40            RowData {
41                time_short,
42                action_label: format!("{arrow} {action_type}"),
43                action_color,
44                process_name: ev.process.name.clone(),
45                pid: ev.process.pid.to_string(),
46                provider: ev.provider().unwrap_or("-").to_string(),
47                dest: ev.sni().unwrap_or("-").to_string(),
48                bytes_str: if bytes > 0 {
49                    format_bytes(bytes)
50                } else {
51                    "-".into()
52                },
53                policy: ev.policy.as_deref().unwrap_or("-").to_string(),
54                policy_str: ev.policy.clone(),
55                is_pii: ev.pii_detected(),
56            }
57        })
58        .collect();
59
60    let row_count = rows.len();
61    let row_height = 28.0;
62    let mut clicked_row: Option<usize> = None;
63
64    let mut table = TableBuilder::new(ui)
65        .striped(false)
66        .cell_layout(egui::Layout::left_to_right(egui::Align::Center))
67        .column(Column::exact(90.0)) // Time
68        .column(Column::exact(130.0)) // Action
69        .column(Column::remainder().at_least(80.0)) // Process
70        .column(Column::exact(50.0)) // PID
71        .column(Column::remainder().at_least(80.0)) // Provider
72        .column(Column::remainder().at_least(100.0)) // Destination
73        .column(Column::exact(60.0)) // Bytes
74        .column(Column::exact(55.0)) // Policy
75        .min_scrolled_height(0.0);
76
77    // Scroll the table viewport to keep the selected row visible
78    if let Some(sel) = selected {
79        table = table.scroll_to_row(sel, None);
80    }
81
82    table = table.sense(egui::Sense::click());
83
84    table
85        .header(row_height, |mut header| {
86            let cols = [
87                "TIME",
88                "ACTION",
89                "PROCESS",
90                "PID",
91                "PROVIDER",
92                "DESTINATION",
93                "BYTES",
94                "POLICY",
95            ];
96            for col in cols {
97                header.col(|ui| {
98                    // Header background
99                    ui.painter()
100                        .rect_filled(ui.max_rect(), 0.0, style::SURFACE_2);
101                    ui.colored_label(
102                        style::DIM_TEXT,
103                        egui::RichText::new(col).size(10.0).strong(),
104                    );
105                });
106            }
107        })
108        .body(|body| {
109            body.rows(row_height, row_count, |mut row| {
110                let idx = row.index();
111                let rd = &rows[idx];
112                let is_selected = selected == Some(idx);
113
114                if is_selected {
115                    row.set_selected(true);
116                }
117
118                row.col(|ui| {
119                    paint_row_bg(ui, rd.is_pii, is_selected);
120                    ui.colored_label(style::DIM_TEXT, &rd.time_short);
121                });
122                row.col(|ui| {
123                    paint_row_bg(ui, rd.is_pii, is_selected);
124                    style::action_pill(ui, &rd.action_label, rd.action_color);
125                });
126                row.col(|ui| {
127                    paint_row_bg(ui, rd.is_pii, is_selected);
128                    ui.colored_label(style::NORMAL_TEXT, &rd.process_name);
129                });
130                row.col(|ui| {
131                    paint_row_bg(ui, rd.is_pii, is_selected);
132                    ui.colored_label(style::DIM_TEXT, &rd.pid);
133                });
134                row.col(|ui| {
135                    paint_row_bg(ui, rd.is_pii, is_selected);
136                    ui.colored_label(style::NORMAL_TEXT, &rd.provider);
137                });
138                row.col(|ui| {
139                    paint_row_bg(ui, rd.is_pii, is_selected);
140                    ui.colored_label(style::DIM_TEXT, &rd.dest);
141                });
142                row.col(|ui| {
143                    paint_row_bg(ui, rd.is_pii, is_selected);
144                    ui.colored_label(style::NORMAL_TEXT, &rd.bytes_str);
145                });
146                row.col(|ui| {
147                    paint_row_bg(ui, rd.is_pii, is_selected);
148                    if rd.policy != "-" {
149                        style::policy_pill(ui, &rd.policy, rd.policy_str.as_deref());
150                    } else {
151                        ui.colored_label(style::DIM_TEXT, "-");
152                    }
153                });
154
155                if row.response().clicked() {
156                    clicked_row = Some(idx);
157                }
158            });
159        });
160
161    // Apply click outside the borrow
162    if let Some(idx) = clicked_row {
163        app.selected_index = Some(idx);
164        app.auto_scroll = false;
165    }
166}

168fn paint_row_bg(ui: &mut egui::Ui, is_pii: bool, is_selected: bool) {
169    let rect = ui.max_rect();
170    if is_selected {
171        ui.painter().rect_filled(rect, 0.0, style::SELECTED_BG);
172        // Left accent border for selected row
173        let accent_rect =
174            egui::Rect::from_min_max(rect.min, egui::pos2(rect.left() + 3.0, rect.bottom()));
175        ui.painter()
176            .rect_filled(accent_rect, 0.0, style::ACCENT_COLOR);
177    } else if is_pii {
178        ui.painter().rect_filled(rect, 0.0, style::PII_ROW_BG);
179        // Left red border for PII rows
180        let pii_rect =
181            egui::Rect::from_min_max(rect.min, egui::pos2(rect.left() + 3.0, rect.bottom()));
182        ui.painter().rect_filled(pii_rect, 0.0, style::PII_COLOR);
183    } else if ui.rect_contains_pointer(rect) {
184        ui.painter().rect_filled(rect, 0.0, style::HOVER_BG);
185    }
186}
