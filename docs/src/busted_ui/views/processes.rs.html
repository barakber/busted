3pub fn show(ui: &mut egui::Ui, app: &mut BustedApp) {
4    ui.heading("Processes Communicating with LLM APIs");
5    ui.separator();
6
7    // Only show processes that have communicated with LLM providers
8    let mut llm_processes: Vec<_> = app
9        .process_info
10        .values()
11        .filter(|p| !p.providers.is_empty())
12        .collect();
13
14    if llm_processes.is_empty() {
15        ui.label("No processes have communicated with LLM APIs yet.");
16        return;
17    }
18
19    llm_processes.sort_by_key(|b| std::cmp::Reverse(b.event_count));
20
21    egui::Grid::new("process_grid")
22        .num_columns(6)
23        .spacing([20.0, 8.0])
24        .striped(true)
25        .show(ui, |ui| {
26            ui.strong("PID");
27            ui.strong("Process");
28            ui.strong("UID");
29            ui.strong("Events");
30            ui.strong("Bytes");
31            ui.strong("Providers");
32            ui.end_row();
33
34            for proc in &llm_processes {
35                ui.label(proc.pid.to_string());
36                ui.label(&proc.name);
37                ui.label(proc.uid.to_string());
38                ui.label(proc.event_count.to_string());
39                ui.label(format_bytes(proc.bytes_total));
40                let providers: Vec<_> = proc.providers.iter().map(|s| s.as_str()).collect();
41                ui.label(providers.join(", "));
42                ui.end_row();
43            }
44        });
45
46    ui.separator();
47    ui.label(format!(
48        "{} unique processes detected communicating with LLM APIs",
49        llm_processes.len()
50    ));
51}
52
53fn format_bytes(b: u64) -> String {
54    if b == 0 {
55        "0 B".to_string()
56    } else if b < 1024 {
57        format!("{} B", b)
58    } else if b < 1024 * 1024 {
59        format!("{:.1} KB", b as f64 / 1024.0)
60    } else {
61        format!("{:.1} MB", b as f64 / (1024.0 * 1024.0))
62    }
63}
