1use crate::app::BustedApp;
2
3pub fn show(ui: &mut egui::Ui, app: &mut BustedApp) {
4    ui.heading("Policy Configuration");
5    ui.separator();
6
7    ui.label(
8        "Configure allow/deny/audit rules for LLM API access. \
9         Note: Enforcement requires the agent to be running with --enforce flag. \
10         Kprobe-based enforcement is audit-only; full blocking requires TC/XDP/LSM (future work).",
11    );
12
13    ui.separator();
14
15    // Existing rules
16    if app.policy_rules.is_empty() {
17        ui.label("No policy rules configured.");
18    } else {
19        let mut to_remove = None;
20
21        egui::Grid::new("policy_rules_grid")
22            .num_columns(4)
23            .spacing([20.0, 8.0])
24            .striped(true)
25            .show(ui, |ui| {
26                ui.strong("Pattern");
27                ui.strong("Action");
28                ui.strong("Enabled");
29                ui.strong("");
30                ui.end_row();
31
32                for (i, rule) in app.policy_rules.iter_mut().enumerate() {
33                    ui.label(&rule.pattern);
34                    ui.label(&rule.action);
35                    ui.checkbox(&mut rule.enabled, "");
36                    if ui.button("Remove").clicked() {
37                        to_remove = Some(i);
38                    }
39                    ui.end_row();
40                }
41            });
42
43        if let Some(idx) = to_remove {
44            app.policy_rules.remove(idx);
45        }
46    }
47
48    ui.separator();
49    ui.heading("Add Rule");
50
51    ui.horizontal(|ui| {
52        ui.label("Pattern (process name or provider):");
53        ui.text_edit_singleline(&mut app.new_rule_pattern);
54    });
55
56    ui.horizontal(|ui| {
57        ui.label("Action:");
58        egui::ComboBox::from_id_salt("new_rule_action")
59            .selected_text(&app.new_rule_action)
60            .show_ui(ui, |ui| {
61                ui.selectable_value(&mut app.new_rule_action, "allow".to_string(), "Allow");
62                ui.selectable_value(&mut app.new_rule_action, "deny".to_string(), "Deny");
63                ui.selectable_value(&mut app.new_rule_action, "audit".to_string(), "Audit");
64            });
65    });
66
67    if ui.button("Add Rule").clicked() && !app.new_rule_pattern.is_empty() {
68        app.policy_rules.push(crate::app::PolicyRule {
69            pattern: app.new_rule_pattern.clone(),
70            action: app.new_rule_action.clone(),
71            enabled: true,
72        });
73        app.new_rule_pattern.clear();
74    }
75}
