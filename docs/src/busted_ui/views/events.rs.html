5/// Categorize an event for the Action filter dropdown.
6fn event_action_category(e: &ProcessedEvent) -> &'static str {
7    if e.mcp_method.is_some() {
8        return "MCP";
9    }
10    match e.event_type.as_str() {
11        "TLS_DATA_WRITE" => "LLM Request",
12        "TLS_DATA_READ" => "LLM Response",
13        _ => "Network",
14    }
15}

17/// Build the compact action string for the Action column.
18fn action_label(e: &ProcessedEvent) -> String {
19    if let Some(ref mcp) = e.mcp_method {
20        let cat = e.mcp_category.as_deref().unwrap_or("");
21        if cat.is_empty() {
22            format!("MCP {}", mcp)
23        } else {
24            format!("MCP {} ({})", mcp, cat)
25        }
26    } else {
27        let provider = e
28            .llm_provider
29            .as_deref()
30            .or(e.provider.as_deref())
31            .unwrap_or("");
32        let model = e.llm_model.as_deref().unwrap_or("");
33        match (provider, model) {
34            ("", "") => "-".to_string(),
35            (p, "") => p.to_string(),
36            (p, m) => format!("{} {}", p, m),
37        }
38    }
39}

41/// Direction arrow for the Dir column.
42fn direction_arrow(e: &ProcessedEvent) -> &'static str {
43    match e.event_type.as_str() {
44        "TLS_DATA_WRITE" => "\u{00bb}", // »
45        "TLS_DATA_READ" => "\u{00ab}",  // «
46        _ => "",
47    }
48}

50/// Compact flags string.
51fn flags_label(e: &ProcessedEvent) -> String {
52    let mut parts = Vec::new();
53    if e.pii_detected == Some(true) {
54        parts.push("PII");
55    }
56    if e.llm_stream == Some(true) {
57        parts.push("STR");
58    }
59    parts.join(" ")
60}

62/// Truncate a string at a character boundary.
63fn truncate_chars(s: &str, max: usize) -> String {
64    match s.char_indices().nth(max) {
65        Some((idx, _)) => format!("{}...", &s[..idx]),
66        None => s.to_string(),
67    }
68}

70pub fn show(ui: &mut egui::Ui, app: &mut BustedApp) {
71    // Filters
72    ui.horizontal(|ui| {
73        ui.label("Action:");
74        egui::ComboBox::from_id_salt("filter_action")
75            .selected_text(if app.filter_action.is_empty() {
76                "All"
77            } else {
78                &app.filter_action
79            })
80            .show_ui(ui, |ui| {
81                ui.selectable_value(&mut app.filter_action, String::new(), "All");
82                for action in &["LLM Request", "LLM Response", "MCP", "Network"] {
83                    ui.selectable_value(&mut app.filter_action, action.to_string(), *action);
84                }
85            });
86
87        ui.label("Provider:");
88        egui::ComboBox::from_id_salt("filter_provider")
89            .selected_text(if app.filter_provider.is_empty() {
90                "All"
91            } else {
92                &app.filter_provider
93            })
94            .show_ui(ui, |ui| {
95                ui.selectable_value(&mut app.filter_provider, String::new(), "All");
96                for provider in app.provider_stats.keys() {
97                    ui.selectable_value(
98                        &mut app.filter_provider,
99                        provider.clone(),
100                        provider.as_str(),
101                    );
102                }
103            });
104
105        ui.label("Process:");
106        ui.text_edit_singleline(&mut app.filter_process);
107
108        ui.checkbox(&mut app.show_network_events, "Network");
109        ui.checkbox(&mut app.auto_scroll, "Auto-scroll");
110    });
111
112    ui.separator();
113
114    // Filtered events — collect indices into app.events
115    let filtered: Vec<usize> = app
116        .events
117        .iter()
118        .enumerate()
119        .filter(|(_, e)| {
120            // Hide non-TLS events unless show_network_events is on
121            if !app.show_network_events && !e.event_type.starts_with("TLS_DATA_") {
122                return false;
123            }
124            // Action filter
125            if !app.filter_action.is_empty() && event_action_category(e) != app.filter_action {
126                return false;
127            }
128            // Provider filter
129            if !app.filter_provider.is_empty() {
130                match &e.provider {
131                    Some(p) if p == &app.filter_provider => {}
132                    _ => return false,
133                }
134            }
135            // Process name filter
136            if !app.filter_process.is_empty()
137                && !e
138                    .process_name
139                    .to_lowercase()
140                    .contains(&app.filter_process.to_lowercase())
141            {
142                return false;
143            }
144            true
145        })
146        .map(|(i, _)| i)
147        .collect();
148
149    let row_count = filtered.len();
150
151    let table = TableBuilder::new(ui)
152        .striped(true)
153        .resizable(true)
154        .sense(egui::Sense::click())
155        .cell_layout(egui::Layout::left_to_right(egui::Align::Center))
156        .column(Column::auto().at_least(80.0)) // Time
157        .column(Column::auto().at_least(110.0)) // Process
158        .column(Column::exact(30.0)) // Dir
159        .column(Column::auto().at_least(200.0)) // Action
160        .column(Column::auto().at_least(120.0)) // SDK
161        .column(Column::remainder().at_least(150.0)) // Content
162        .column(Column::auto().at_least(55.0)) // Policy
163        .column(Column::auto().at_least(70.0)) // Flags
164        .min_scrolled_height(0.0);
165
166    let table = if app.auto_scroll {
167        table.scroll_to_row(row_count.saturating_sub(1), None)
168    } else {
169        table
170    };
171
172    table
173        .header(20.0, |mut header| {
174            header.col(|ui| {
175                ui.strong("Time");
176            });
177            header.col(|ui| {
178                ui.strong("Process");
179            });
180            header.col(|ui| {
181                ui.strong("Dir");
182            });
183            header.col(|ui| {
184                ui.strong("Action");
185            });
186            header.col(|ui| {
187                ui.strong("SDK");
188            });
189            header.col(|ui| {
190                ui.strong("Content");
191            });
192            header.col(|ui| {
193                ui.strong("Policy");
194            });
195            header.col(|ui| {
196                ui.strong("Flags");
197            });
198        })
199        .body(|body| {
200            body.rows(18.0, row_count, |mut row| {
201                let event_idx = filtered[row.index()];
202                let selected = app.selected_event_idx == Some(event_idx);
203                row.set_selected(selected);
204
205                let event = &app.events[event_idx];
206
207                row.col(|ui| {
208                    // Show just HH:MM:SS (trim millis for compactness)
209                    let ts = if event.timestamp.len() > 8 {
210                        &event.timestamp[..8]
211                    } else {
212                        &event.timestamp
213                    };
214                    ui.label(ts);
215                });
216                row.col(|ui| {
217                    ui.label(format!("{} ({})", event.process_name, event.pid));
218                });
219                row.col(|ui| {
220                    ui.label(direction_arrow(event));
221                });
222                row.col(|ui| {
223                    ui.label(action_label(event));
224                });
225                row.col(|ui| {
226                    ui.label(event.agent_sdk.as_deref().unwrap_or("-"));
227                });
228                row.col(|ui| {
229                    let content = event.llm_user_message.as_deref().unwrap_or("-");
230                    ui.label(truncate_chars(content, 60));
231                });
232                row.col(|ui| {
233                    let policy = event.policy.as_deref().unwrap_or("-");
234                    let color = match policy {
235                        "allow" => egui::Color32::from_rgb(80, 200, 80),
236                        "audit" => egui::Color32::from_rgb(220, 180, 40),
237                        "deny" => egui::Color32::from_rgb(220, 60, 60),
238                        _ => ui.visuals().text_color(),
239                    };
240                    ui.colored_label(color, policy);
241                });
242                row.col(|ui| {
243                    let flags = flags_label(event);
244                    if flags.contains("PII") {
245                        ui.colored_label(egui::Color32::from_rgb(220, 60, 60), &flags);
246                    } else {
247                        ui.label(&flags);
248                    }
249                });
250
251                // Handle row click — toggle selection
252                if row.response().clicked() {
253                    if app.selected_event_idx == Some(event_idx) {
254                        app.selected_event_idx = None;
255                    } else {
256                        app.selected_event_idx = Some(event_idx);
257                    }
258                }
259            });
260        });
261}

263/// Render the detail panel for a selected event.
264pub fn show_detail_panel(ui: &mut egui::Ui, event: &ProcessedEvent) {
265    egui::ScrollArea::vertical()
266        .id_salt("detail_scroll")
267        .show(ui, |ui| {
268            ui.horizontal(|ui| {
269                ui.strong("Event Detail");
270                ui.label(format!("  {}  {}", event.event_type, event.timestamp));
271            });
272            ui.separator();
273
274            // Key-value grid
275            egui::Grid::new("detail_grid")
276                .num_columns(2)
277                .spacing([12.0, 4.0])
278                .show(ui, |ui| {
279                    detail_row(
280                        ui,
281                        "Process",
282                        &format!(
283                            "{} (PID: {}, UID: {})",
284                            event.process_name, event.pid, event.uid
285                        ),
286                    );
287                    detail_row(
288                        ui,
289                        "Provider",
290                        event
291                            .llm_provider
292                            .as_deref()
293                            .or(event.provider.as_deref())
294                            .unwrap_or("-"),
295                    );
296                    if let Some(ref model) = event.llm_model {
297                        detail_row(ui, "Model", model);
298                    }
299                    if let Some(ref endpoint) = event.llm_endpoint {
300                        detail_row(ui, "Endpoint", endpoint);
301                    }
302                    if let Some(ref sdk) = event.agent_sdk {
303                        detail_row(ui, "SDK", sdk);
304                    }
305                    if let Some(ref method) = event.mcp_method {
306                        detail_row(ui, "MCP Method", method);
307                    }
308                    if let Some(ref cat) = event.mcp_category {
309                        detail_row(ui, "MCP Category", cat);
310                    }
311                    if let Some(ref policy) = event.policy {
312                        detail_row(ui, "Policy", policy);
313                    }
314                    if let Some(pii) = event.pii_detected {
315                        detail_row(ui, "PII Detected", if pii { "YES" } else { "no" });
316                    }
317                    if let Some(stream) = event.llm_stream {
318                        detail_row(ui, "Streaming", if stream { "yes" } else { "no" });
319                    }
320                    if let Some(conf) = event.classifier_confidence {
321                        detail_row(ui, "Confidence", &format!("{:.0}%", conf * 100.0));
322                    }
323                    // Identity
324                    if let Some(ref narrative) = event.identity_narrative {
325                        detail_row(ui, "Identity", narrative);
326                    }
327                    if let Some(ref timeline) = event.identity_timeline {
328                        detail_row(ui, "Timeline", timeline);
329                    }
330                    // Network (secondary)
331                    detail_row(
332                        ui,
333                        "Network",
334                        &format!(
335                            "{}:{} -> {}:{} ({} bytes)",
336                            event.src_ip, event.src_port, event.dst_ip, event.dst_port, event.bytes,
337                        ),
338                    );
339                    if let Some(ref sni) = event.sni {
340                        detail_row(ui, "SNI", sni);
341                    }
342                    if !event.container_id.is_empty() {
343                        detail_row(ui, "Container", &event.container_id);
344                    }
345                });
346
347            // Full content sections
348            if let Some(ref msg) = event.llm_user_message {
349                ui.separator();
350                ui.strong("User Message");
351                egui::ScrollArea::vertical()
352                    .id_salt("user_msg_scroll")
353                    .max_height(100.0)
354                    .show(ui, |ui| {
355                        ui.label(msg);
356                    });
357            }
358            if let Some(ref prompt) = event.llm_system_prompt {
359                ui.separator();
360                ui.strong("System Prompt");
361                egui::ScrollArea::vertical()
362                    .id_salt("sys_prompt_scroll")
363                    .max_height(100.0)
364                    .show(ui, |ui| {
365                        ui.label(prompt);
366                    });
367            }
368        });
369}
