4pub fn show(ui: &mut egui::Ui, app: &mut BustedApp) {
5    // Filters
6    ui.horizontal(|ui| {
7        ui.label("Filter:");
8        ui.label("Provider:");
9        egui::ComboBox::from_id_salt("filter_provider")
10            .selected_text(if app.filter_provider.is_empty() {
11                "All"
12            } else {
13                &app.filter_provider
14            })
15            .show_ui(ui, |ui| {
16                ui.selectable_value(&mut app.filter_provider, String::new(), "All");
17                for provider in app.provider_stats.keys() {
18                    ui.selectable_value(
19                        &mut app.filter_provider,
20                        provider.clone(),
21                        provider.as_str(),
22                    );
23                }
24            });
25
26        ui.label("Type:");
27        egui::ComboBox::from_id_salt("filter_event_type")
28            .selected_text(if app.filter_event_type.is_empty() {
29                "All"
30            } else {
31                &app.filter_event_type
32            })
33            .show_ui(ui, |ui| {
34                ui.selectable_value(&mut app.filter_event_type, String::new(), "All");
35                for t in &[
36                    "TCP_CONNECT",
37                    "DATA_SENT",
38                    "DATA_RECEIVED",
39                    "CONNECTION_CLOSED",
40                    "DNS_QUERY",
41                    "TLS_DATA_WRITE",
42                    "TLS_DATA_READ",
43                ] {
44                    ui.selectable_value(&mut app.filter_event_type, t.to_string(), *t);
45                }
46            });
47
48        ui.label("Process:");
49        ui.text_edit_singleline(&mut app.filter_process);
50
51        ui.checkbox(&mut app.auto_scroll, "Auto-scroll");
52    });
53
54    ui.separator();
55
56    // Filtered events
57    let filtered: Vec<usize> = app
58        .events
59        .iter()
60        .enumerate()
61        .filter(|(_, e)| {
62            if !app.filter_provider.is_empty() {
63                match &e.provider {
64                    Some(p) if p == &app.filter_provider => {}
65                    _ => return false,
66                }
67            }
68            if !app.filter_event_type.is_empty() && e.event_type != app.filter_event_type {
69                return false;
70            }
71            if !app.filter_process.is_empty()
72                && !e
73                    .process_name
74                    .to_lowercase()
75                    .contains(&app.filter_process.to_lowercase())
76            {
77                return false;
78            }
79            true
80        })
81        .map(|(i, _)| i)
82        .collect();
83
84    let row_count = filtered.len();
85
86    let table = TableBuilder::new(ui)
87        .striped(true)
88        .resizable(true)
89        .cell_layout(egui::Layout::left_to_right(egui::Align::Center))
90        .column(Column::auto().at_least(80.0)) // Time
91        .column(Column::auto().at_least(90.0)) // Type
92        .column(Column::auto().at_least(50.0)) // PID
93        .column(Column::auto().at_least(80.0)) // Process
94        .column(Column::auto().at_least(110.0)) // Dst IP
95        .column(Column::auto().at_least(50.0)) // Port
96        .column(Column::auto().at_least(70.0)) // Bytes
97        .column(Column::auto().at_least(70.0)) // Provider
98        .column(Column::auto().at_least(80.0)) // Model
99        .column(Column::auto().at_least(100.0)) // SDK
100        .column(Column::auto().at_least(90.0)) // MCP Method
101        .column(Column::auto().at_least(120.0)) // SNI
102        .column(Column::auto().at_least(80.0)) // Container
103        .min_scrolled_height(0.0);
104
105    let table = if app.auto_scroll {
106        table.scroll_to_row(row_count.saturating_sub(1), None)
107    } else {
108        table
109    };
110
111    table
112        .header(20.0, |mut header| {
113            header.col(|ui| {
114                ui.strong("Time");
115            });
116            header.col(|ui| {
117                ui.strong("Type");
118            });
119            header.col(|ui| {
120                ui.strong("PID");
121            });
122            header.col(|ui| {
123                ui.strong("Process");
124            });
125            header.col(|ui| {
126                ui.strong("Dst IP");
127            });
128            header.col(|ui| {
129                ui.strong("Port");
130            });
131            header.col(|ui| {
132                ui.strong("Bytes");
133            });
134            header.col(|ui| {
135                ui.strong("Provider");
136            });
137            header.col(|ui| {
138                ui.strong("Model");
139            });
140            header.col(|ui| {
141                ui.strong("SDK");
142            });
143            header.col(|ui| {
144                ui.strong("MCP Method");
145            });
146            header.col(|ui| {
147                ui.strong("SNI");
148            });
149            header.col(|ui| {
150                ui.strong("Container");
151            });
152        })
153        .body(|body| {
154            body.rows(18.0, row_count, |mut row| {
155                let idx = filtered[row.index()];
156                let event = &app.events[idx];
157                row.col(|ui| {
158                    ui.label(&event.timestamp);
159                });
160                row.col(|ui| {
161                    ui.label(&event.event_type);
162                });
163                row.col(|ui| {
164                    ui.label(event.pid.to_string());
165                });
166                row.col(|ui| {
167                    ui.label(&event.process_name);
168                });
169                row.col(|ui| {
170                    ui.label(&event.dst_ip);
171                });
172                row.col(|ui| {
173                    ui.label(event.dst_port.to_string());
174                });
175                row.col(|ui| {
176                    ui.label(format_bytes(event.bytes));
177                });
178                row.col(|ui| {
179                    ui.label(
180                        event
181                            .llm_provider
182                            .as_deref()
183                            .or(event.provider.as_deref())
184                            .unwrap_or("-"),
185                    );
186                });
187                row.col(|ui| {
188                    ui.label(event.llm_model.as_deref().unwrap_or("-"));
189                });
190                row.col(|ui| {
191                    ui.label(event.agent_sdk.as_deref().unwrap_or("-"));
192                });
193                row.col(|ui| {
194                    ui.label(event.mcp_method.as_deref().unwrap_or("-"));
195                });
196                row.col(|ui| {
197                    ui.label(event.sni.as_deref().unwrap_or("-"));
198                });
199                row.col(|ui| {
200                    let cid = if event.container_id.is_empty() {
201                        "-"
202                    } else {
203                        &event.container_id
204                    };
205                    ui.label(cid);
206                });
207            });
208        });

211fn format_bytes(b: u64) -> String {
212    if b == 0 {
213        "-".to_string()
214    } else if b < 1024 {
215        format!("{} B", b)
216    } else if b < 1024 * 1024 {
217        format!("{:.1} KB", b as f64 / 1024.0)
218    } else {
219        format!("{:.1} MB", b as f64 / (1024.0 * 1024.0))
220    }
