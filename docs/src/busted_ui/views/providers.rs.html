3pub fn show(ui: &mut egui::Ui, app: &mut BustedApp) {
4    ui.heading("LLM Provider Summary");
5    ui.separator();
6
7    if app.provider_stats.is_empty() {
8        ui.label("No LLM provider traffic detected yet.");
9        return;
10    }
11
12    // Sort providers by event count (descending)
13    let mut providers: Vec<_> = app.provider_stats.iter().collect();
14    providers.sort_by_key(|b| std::cmp::Reverse(b.1.event_count));
15
16    let max_events = providers
17        .first()
18        .map(|(_, s)| s.event_count)
19        .unwrap_or(1)
20        .max(1) as f32;
21
22    egui::Grid::new("provider_grid")
23        .num_columns(5)
24        .spacing([20.0, 8.0])
25        .striped(true)
26        .show(ui, |ui| {
27            ui.strong("Provider");
28            ui.strong("Events");
29            ui.strong("Bytes");
30            ui.strong("Processes");
31            ui.strong("Activity");
32            ui.end_row();
33
34            for (name, stats) in &providers {
35                ui.label(*name);
36                ui.label(stats.event_count.to_string());
37                ui.label(format_bytes(stats.bytes_total));
38                ui.label(stats.processes.len().to_string());
39
40                // Simple bar chart
41                let fraction = stats.event_count as f32 / max_events;
42                let (rect, _) =
43                    ui.allocate_exact_size(egui::vec2(200.0, 16.0), egui::Sense::hover());
44                let bar_rect = egui::Rect::from_min_size(
45                    rect.min,
46                    egui::vec2(rect.width() * fraction, rect.height()),
47                );
48                ui.painter()
49                    .rect_filled(bar_rect, 2.0, provider_color(name));
50                ui.painter().rect_stroke(
51                    rect,
52                    2.0,
53                    egui::Stroke::new(1.0, egui::Color32::GRAY),
54                    egui::StrokeKind::Outside,
55                );
56
57                ui.end_row();
58            }
59        });
60
61    ui.separator();
62
63    // Total summary
64    let total_events: u64 = app.provider_stats.values().map(|s| s.event_count).sum();
65    let total_bytes: u64 = app.provider_stats.values().map(|s| s.bytes_total).sum();
66    ui.label(format!(
67        "Total: {} events | {} transferred across {} providers",
68        total_events,
69        format_bytes(total_bytes),
70        app.provider_stats.len(),
71    ));
72}

74fn provider_color(name: &str) -> egui::Color32 {
75    match name {
76        "OpenAI" => egui::Color32::from_rgb(116, 184, 117),
77        "Anthropic" => egui::Color32::from_rgb(204, 133, 80),
78        "Google" => egui::Color32::from_rgb(66, 133, 244),
79        "Azure" => egui::Color32::from_rgb(0, 120, 212),
80        "AWS Bedrock" => egui::Color32::from_rgb(255, 153, 0),
81        "Cohere" => egui::Color32::from_rgb(209, 98, 133),
82        "HuggingFace" => egui::Color32::from_rgb(255, 216, 0),
83        _ => egui::Color32::GRAY,
84    }
85}

87fn format_bytes(b: u64) -> String {
88    if b == 0 {
89        "0 B".to_string()
90    } else if b < 1024 {
91        format!("{} B", b)
92    } else if b < 1024 * 1024 {
93        format!("{:.1} KB", b as f64 / 1024.0)
94    } else if b < 1024 * 1024 * 1024 {
95        format!("{:.1} MB", b as f64 / (1024.0 * 1024.0))
96    } else {
97        format!("{:.1} GB", b as f64 / (1024.0 * 1024.0 * 1024.0))
98    }
