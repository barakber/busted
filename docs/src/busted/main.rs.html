99/// Known subnet prefixes for LLM providers (covers CDN/anycast rotation).
100const LLM_SUBNETS: &[([u8; 2], &str)] = &[
101    // Anthropic (160.79.x.x)
102    ([160, 79], "Anthropic"),
103    // Cloudflare ranges used by OpenAI (104.18.x.x, 172.66.x.x, 162.159.x.x)
104    ([104, 18], "OpenAI"),
105    ([172, 66], "OpenAI"),
106    ([162, 159], "OpenAI"),
107];

109/// Known /32 IPv6 prefixes for LLM providers.
110const LLM_SUBNETS_V6: &[([u8; 4], &str)] = &[
111    // Anthropic 2607:6bc0::/32
112    ([0x26, 0x07, 0x6b, 0xc0], "Anthropic"),
113    // Google Cloud 2600:1901::/32 (used by Anthropic infrastructure)
114    ([0x26, 0x00, 0x19, 0x01], "Anthropic"),
115];

117/// Resolve all LLM endpoints to IPs and return the map.
118fn resolve_provider_ips() -> HashMap<IpAddr, &'static str> {
119    let mut m = HashMap::new();
120    for &(hostname, provider) in LLM_ENDPOINTS {
121        let addr = format!("{}:443", hostname);
122        match addr.to_socket_addrs() {
123            Ok(addrs) => {
124                for a in addrs {
125                    info!("  {} -> {} ({})", hostname, a.ip(), provider);
126                    m.insert(a.ip(), provider);
127                }
128            }
129            Err(e) => {
130                warn!("Failed to resolve {}: {}", hostname, e);
131            }
132        }
133    }
134    info!("Resolved {} LLM provider IPs", m.len());
135    m
136}

138fn classify_llm_provider(
139    ip: &IpAddr,
140    dport: u16,
141    provider_map: &HashMap<IpAddr, &'static str>,
142) -> Option<&'static str> {
143    if dport != 443 {
144        return None;
145    }
146
147    // 1. Exact IP match from DNS resolution
148    if let Some(provider) = provider_map.get(ip).copied() {
149        return Some(provider);
150    }
151
152    // 2. Subnet prefix match for CDN/anycast ranges
153    match ip {
154        IpAddr::V4(v4) => {
155            let octets = v4.octets();
156            for &(prefix, provider) in LLM_SUBNETS {
157                if octets[0] == prefix[0] && octets[1] == prefix[1] {
158                    return Some(provider);
159                }
160            }
161        }
162        IpAddr::V6(v6) => {
163            let octets = v6.octets();
164            for &(prefix, provider) in LLM_SUBNETS_V6 {
165                if octets[0] == prefix[0]
166                    && octets[1] == prefix[1]
167                    && octets[2] == prefix[2]
168                    && octets[3] == prefix[3]
169                {
170                    return Some(provider);
171                }
172            }
173        }
174    }
175
176    None
177}

183fn resolve_container_id(pid: u32, cache: &mut HashMap<u32, String>) -> String {
184    if let Some(cached) = cache.get(&pid) {
185        return cached.clone();
186    }
187
188    let container_id = resolve_container_id_from_proc(pid);
189
190    // LRU-style: clear cache if too large
191    if cache.len() > 10_000 {
192        cache.clear();
193    }
194    cache.insert(pid, container_id.clone());
195    container_id
196}
197
198fn resolve_container_id_from_proc(pid: u32) -> String {
199    let path = format!("/proc/{}/cgroup", pid);
200    let content = match std::fs::read_to_string(&path) {
201        Ok(c) => c,
202        Err(_) => return String::new(),
203    };
204
205    // Match Docker or containerd container IDs
206    let re = Regex::new(r"(?:docker-|cri-containerd-)([a-f0-9]{64})\.scope").unwrap();
207    if let Some(caps) = re.captures(&content) {
208        if let Some(m) = caps.get(1) {
209            return m.as_str()[..12].to_string(); // Short container ID
210        }
211    }
212
213    // Also try bare hex IDs in cgroup path (common in k8s)
214    let re2 = Regex::new(r"/([a-f0-9]{64})$").unwrap();
215    for line in content.lines() {
216        if let Some(caps) = re2.captures(line) {
217            if let Some(m) = caps.get(1) {
218                return m.as_str()[..12].to_string();
219            }
220        }
221    }
222
223    String::new()
224}

230struct PidStats {
231    first_seen: Instant,
232    event_count: u64,
233    bytes_total: u64,
234}
235
236impl PidStats {
237    fn new() -> Self {
238        Self {
239            first_seen: Instant::now(),
240            event_count: 0,
241            bytes_total: 0,
242        }
243    }
244
245    fn request_rate(&self) -> f64 {
246        let elapsed = self.first_seen.elapsed().as_secs_f64();
247        if elapsed < 0.001 {
248            0.0
249        } else {
250            self.event_count as f64 / elapsed
251        }
252    }
253}

366#[allow(clippy::too_many_arguments)]
367async fn handle_event(
368    event: NetworkEvent,
369    tx: &broadcast::Sender<ProcessedEvent>,
370    identity_map: &Arc<Mutex<AyaHashMap<MapData, u32, AgentIdentity>>>,
371    policy_map: &Arc<Mutex<AyaHashMap<MapData, u32, u8>>>,
372    provider_map: &Arc<RwLock<HashMap<IpAddr, &'static str>>>,
373    enforce: bool,
374    container_cache: &mut HashMap<u32, String>,
375    pid_stats: &mut HashMap<u32, PidStats>,
376    #[cfg(feature = "k8s")] k8s_cache: &Arc<RwLock<HashMap<String, k8s::PodMetadata>>>,
377    #[cfg(feature = "ml")] ml_classifier: &mut ml::MlClassifier,
378    #[cfg(feature = "tls")] sni_cache: &tls::SniCache,
379    #[cfg(feature = "prometheus")] llm_pids: &mut HashSet<u32>,
380    #[cfg(feature = "prometheus")] unique_providers: &mut HashSet<String>,
381) {
382    let dst_ip = event.dest_ip();
383
384    // Try SNI-based classification first (more reliable than IP)
385    #[cfg(feature = "tls")]
386    let sni_hostname: Option<&str> = sni_cache.get(event.pid);
387
388    let pmap = provider_map.read().await;
389    #[cfg(feature = "tls")]
390    let provider = {
391        if event.dport == 443 {
392            if let Some(sni) = sni_hostname {
393                tls::classify_by_sni(sni)
394                    .or_else(|| classify_llm_provider(&dst_ip, event.dport, &pmap))
395            } else {
396                classify_llm_provider(&dst_ip, event.dport, &pmap)
397            }
398        } else {
399            classify_llm_provider(&dst_ip, event.dport, &pmap)
400        }
401    };
402    #[cfg(not(feature = "tls"))]
403    let provider = classify_llm_provider(&dst_ip, event.dport, &pmap);
404    drop(pmap);
405
406    let policy_str = if provider.is_some() && enforce {
407        if let Ok(mut pmap) = policy_map.try_lock() {
408            // With LSM: write Deny (1) to block; without LSM: Audit (2)
409            let _ = pmap.insert(event.pid, 1, 0);
410        }
411        Some("deny")
412    } else if provider.is_some() {
413        if let Ok(mut pmap) = policy_map.try_lock() {
414            let _ = pmap.insert(event.pid, 2, 0);
415        }
416        Some("audit")
417    } else {
418        None
419    };
420
421    // Record policy decision metric
422    #[cfg(feature = "prometheus")]
423    if let Some(decision) = policy_str {
424        metrics::record_policy_decision(decision);
425    }
426
427    // Store identity for processes communicating with LLM providers
428    if provider.is_some() {
429        if let Ok(mut imap) = identity_map.try_lock() {
430            let mut identity = AgentIdentity::new();
431            identity.pid = event.pid;
432            identity.uid = event.uid;
433            identity.comm = event.comm;
434            identity.container_id = event.container_id;
435            identity.created_at_ns = event.timestamp_ns;
436            let _ = imap.insert(event.pid, identity, 0);
437        }
438    }
439
440    // Resolve container ID from /proc
441    let container_id = resolve_container_id(event.pid, container_cache);
442
443    // Update per-PID traffic stats
444    let stats = pid_stats.entry(event.pid).or_insert_with(PidStats::new);
445    stats.event_count += 1;
446    stats.bytes_total += event.bytes;
447    let request_rate = stats.request_rate();
448    let session_bytes = stats.bytes_total;
449
450    // Flag high-rate port-443 traffic as possible LLM even without IP match
451    if provider.is_none() && event.dport == 443 && request_rate > 10.0 {
452        debug!(
453            "High-rate TLS traffic from PID {} ({}): {:.1} events/sec",
454            event.pid,
455            event.process_name(),
456            request_rate
457        );
458    }
459
460    let mut processed = events::from_network_event(&event, provider, policy_str);
461    // Override container_id with the one resolved from /proc (more reliable)
462    if !container_id.is_empty() {
463        processed.container_id = container_id;
464    }
465    processed.request_rate = Some(request_rate);
466    processed.session_bytes = Some(session_bytes);
467
468    // Attach SNI hostname if available
469    #[cfg(feature = "tls")]
470    {
471        processed.sni = sni_hostname.map(|s| s.to_string());
472    }
473
474    // Enrich with Kubernetes pod metadata if available
475    #[cfg(feature = "k8s")]
476    {
477        let k8s_map = k8s_cache.read().await;
478        if let Some(meta) = k8s::resolve_pod_metadata(&processed.container_id, &k8s_map) {
479            processed.pod_name = Some(meta.pod_name);
480            processed.pod_namespace = Some(meta.namespace);
481            processed.service_account = Some(meta.service_account);
482        }
483    }
484
485    // ML behavioral classification
486    #[cfg(feature = "ml")]
487    {
488        let behavior = ml_classifier.process_event(&event, provider);
489        if let Some(ref b) = behavior {
490            // If ML detects LLM traffic with high confidence but IP match missed it,
491            // promote the ML prediction to the provider field.
492            if b.is_novel && b.confidence > 0.85 && processed.provider.is_none() {
493                if let ml::BehaviorClass::LlmApi(ref p) = b.class {
494                    processed.provider = Some(format!("{} (ML)", p));
495                }
496            }
497            processed.ml_confidence = Some(b.confidence);
498            processed.behavior_class = Some(b.class.to_string());
499            processed.cluster_id = Some(b.cluster_id);
500            if let ml::BehaviorClass::LlmApi(ref p) = b.class {
501                processed.ml_provider = Some(p.clone());
502            }
503
504            #[cfg(feature = "prometheus")]
505            metrics::record_ml_classification(&b.class.to_string());
506        }
507    }
508
509    // Record event metrics (all events, not just interesting ones)
510    #[cfg(feature = "prometheus")]
511    metrics::record_event(
512        &processed.event_type,
513        processed.provider.as_deref(),
514        processed.bytes,
515    );
516
517    // Only broadcast interesting events (LLM/AI-related)
518    // Non-interesting traffic is only visible at debug log level
519    if processed.provider.is_some() {
520        // Track unique PIDs and providers for metrics
521        #[cfg(feature = "prometheus")]
522        {
523            llm_pids.insert(processed.pid);
524            metrics::set_active_pids(llm_pids.len());
525            if let Some(ref p) = processed.provider {
526                if unique_providers.insert(p.clone()) {
527                    metrics::set_providers_detected(unique_providers.len());
528                }
529            }
530        }
531        let _ = tx.send(processed);
532    } else {
533        debug!(
534            "[{}] PID: {} ({}) | {}:{} -> {}:{} | {} bytes",
535            processed.event_type,
536            processed.pid,
537            processed.process_name,
538            processed.src_ip,
539            processed.src_port,
540            processed.dst_ip,
541            processed.dst_port,
542            processed.bytes,
543        );
544    }
545}

551async fn cli_output_consumer(mut rx: broadcast::Receiver<ProcessedEvent>, format: String) {
552    loop {
553        match rx.recv().await {
554            Ok(event) => match format.as_str() {
555                "json" => {
556                    let output = EventOutput::from(&event);
557                    if let Ok(json) = serde_json::to_string(&output) {
558                        println!("{}", json);
559                    }
560                }
561                _ => {
562                    // TLS data events: show decrypted payload
563                    if event.event_type.starts_with("TLS_DATA_") {
564                        if let Some(ref payload) = event.tls_payload {
565                            info!(
566                                "[{}] {} | PID: {} ({}) | {} bytes | {}{}\n---\n{}\n---",
567                                event.event_type,
568                                event.timestamp,
569                                event.pid,
570                                event.process_name,
571                                event.bytes,
572                                event.tls_protocol.as_deref().unwrap_or(""),
573                                event
574                                    .tls_details
575                                    .as_ref()
576                                    .map(|d| format!(" ({})", d))
577                                    .unwrap_or_default(),
578                                payload,
579                            );
580                        }
581                    } else {
582                        info!(
583                            "[{}] {} | PID: {} ({}) | UID: {} | {}:{} -> {}:{} | {} bytes | Provider: {}{}{}",
584                            event.event_type,
585                            event.timestamp,
586                            event.pid,
587                            event.process_name,
588                            event.uid,
589                            event.src_ip,
590                            event.src_port,
591                            event.dst_ip,
592                            event.dst_port,
593                            event.bytes,
594                            event.provider.as_deref().unwrap_or("unknown"),
595                            event.policy.as_ref().map(|p| format!(" | Policy: {}", p)).unwrap_or_default(),
596                            if !event.container_id.is_empty() {
597                                format!(" | Container: {}", event.container_id)
598                            } else {
599                                String::new()
600                            },
601                        );
602                    }
603                }
604            },
605            Err(broadcast::error::RecvError::Lagged(n)) => {
606                warn!("CLI consumer lagged, dropped {} events", n);
607            }
608            Err(broadcast::error::RecvError::Closed) => break,
609        }
610    }
611}

617#[tokio::main]
618async fn main() -> Result<()> {
619    let cli = Cli::parse();
620
621    env_logger::Builder::from_env(
622        env_logger::Env::default().default_filter_or(if cli.verbose { "debug" } else { "info" }),
623    )
624    .init();
625
626    // Initialize Prometheus metrics exporter
627    #[cfg(feature = "prometheus")]
628    metrics::init(cli.metrics_port)?;
629
630    // Resolve LLM provider IPs into shared map
631    let provider_map = Arc::new(RwLock::new(resolve_provider_ips()));
632
633    #[cfg(feature = "prometheus")]
634    metrics::set_dns_resolutions(provider_map.read().await.len());
635
636    // Spawn periodic DNS re-resolution task
637    {
638        let pm = provider_map.clone();
639        task::spawn(async move {
640            loop {
641                tokio::time::sleep(Duration::from_secs(60)).await;
642                let new_map = tokio::task::spawn_blocking(resolve_provider_ips)
643                    .await
644                    .unwrap_or_default();
645                let count = new_map.len();
646                *pm.write().await = new_map;
647                #[cfg(feature = "prometheus")]
648                metrics::set_dns_resolutions(count);
649                info!("Re-resolved {} LLM provider IPs", count);
650            }
651        });
652    }
653
654    // Bump memlock rlimit for older kernels
655    let rlim = libc::rlimit {
656        rlim_cur: libc::RLIM_INFINITY,
657        rlim_max: libc::RLIM_INFINITY,
658    };
659    let ret = unsafe { libc::setrlimit(libc::RLIMIT_MEMLOCK, &rlim) };
660    if ret != 0 {
661        warn!("Failed to increase rlimit");
662    }
663
664    // Load eBPF program
665    #[cfg(debug_assertions)]
666    let mut bpf = Ebpf::load(include_bytes_aligned!(
667        "../../target/bpfel-unknown-none/debug/busted-ebpf"
668    ))?;
669    #[cfg(not(debug_assertions))]
670    let mut bpf = Ebpf::load(include_bytes_aligned!(
671        "../../target/bpfel-unknown-none/release/busted-ebpf"
672    ))?;
673
674    if let Err(e) = EbpfLogger::init(&mut bpf) {
675        warn!("Failed to initialize eBPF logger: {}", e);
676    }
677
678    info!("Loading eBPF programs...");
679
680    // Attach kprobes
681    let program: &mut KProbe = bpf.program_mut("tcp_connect").unwrap().try_into()?;
682    program.load()?;
683    program
684        .attach("tcp_connect", 0)
685        .context("Failed to attach tcp_connect")?;
686    info!("Attached to tcp_connect");
687
688    let program: &mut KProbe = bpf.program_mut("tcp_sendmsg").unwrap().try_into()?;
689    program.load()?;
690    program
691        .attach("tcp_sendmsg", 0)
692        .context("Failed to attach tcp_sendmsg")?;
693    info!("Attached to tcp_sendmsg");
694
695    let program: &mut KProbe = bpf.program_mut("tcp_recvmsg").unwrap().try_into()?;
696    program.load()?;
697    program
698        .attach("tcp_recvmsg", 0)
699        .context("Failed to attach tcp_recvmsg")?;
700    info!("Attached to tcp_recvmsg");
701
702    let program: &mut KProbe = bpf.program_mut("tcp_close").unwrap().try_into()?;
703    program.load()?;
704    program
705        .attach("tcp_close", 0)
706        .context("Failed to attach tcp_close")?;
707    info!("Attached to tcp_close");
708
709    let program: &mut KProbe = bpf.program_mut("udp_sendmsg").unwrap().try_into()?;
710    program.load()?;
711    program
712        .attach("udp_sendmsg", 0)
713        .context("Failed to attach udp_sendmsg")?;
714    info!("Attached to udp_sendmsg (DNS probe)");
715
716    // Try to attach LSM hook (optional  requires CONFIG_BPF_LSM and lsm=bpf boot param)
717    match Btf::from_sys_fs() {
718        Ok(btf) => {
719            if let Some(prog) = bpf.program_mut("lsm_socket_connect") {
720                match TryInto::<&mut Lsm>::try_into(prog) {
721                    Ok(lsm) => match lsm.load("socket_connect", &btf) {
722                        Ok(()) => match lsm.attach() {
723                            Ok(_) => info!("LSM socket_connect enforcement attached"),
724                            Err(e) => warn!("LSM attach failed (BPF not in LSM list?): {}", e),
725                        },
726                        Err(e) => warn!("LSM load failed: {}", e),
727                    },
728                    Err(e) => warn!("LSM program type mismatch: {}", e),
729                }
730            }
731        }
732        Err(e) => {
733            warn!("BTF not available, LSM hooks disabled: {}", e);
734        }
735    }
736
737    // Attach TLS uprobes (behind tls feature flag)
738    #[cfg(feature = "tls")]
739    {
740        if let Some(libssl_path) = tls::detect_libssl_path() {
741            // SNI extraction (existing)
742            let prog: &mut UProbe = bpf.program_mut("ssl_ctrl_sni").unwrap().try_into()?;
743            prog.load()?;
744            match prog.attach(Some("SSL_ctrl"), 0, &libssl_path, None) {
745                Ok(_) => info!(
746                    "TLS uprobe attached to SSL_ctrl at {}",
747                    libssl_path.display()
748                ),
749                Err(e) => warn!("Failed to attach TLS uprobe: {}", e),
750            }
751
752            // SSL_write plaintext capture
753            let prog: &mut UProbe = bpf.program_mut("ssl_write_entry").unwrap().try_into()?;
754            prog.load()?;
755            match prog.attach(Some("SSL_write"), 0, &libssl_path, None) {
756                Ok(_) => info!("TLS uprobe attached to SSL_write"),
757                Err(e) => warn!("Failed to attach SSL_write uprobe: {}", e),
758            }
759
760            // SSL_read entry (stash args)
761            let prog: &mut UProbe = bpf.program_mut("ssl_read_entry").unwrap().try_into()?;
762            prog.load()?;
763            match prog.attach(Some("SSL_read"), 0, &libssl_path, None) {
764                Ok(_) => info!("TLS uprobe attached to SSL_read"),
765                Err(e) => warn!("Failed to attach SSL_read uprobe: {}", e),
766            }
767
768            // SSL_read return (read buffer)
769            let prog: &mut UProbe = bpf.program_mut("ssl_read_ret").unwrap().try_into()?;
770            prog.load()?;
771            match prog.attach(Some("SSL_read"), 0, &libssl_path, None) {
772                Ok(_) => info!("TLS uretprobe attached to SSL_read"),
773                Err(e) => warn!("Failed to attach SSL_read uretprobe: {}", e),
774            }
775
776            // SSL_free cleanup
777            let prog: &mut UProbe = bpf.program_mut("ssl_free_cleanup").unwrap().try_into()?;
778            prog.load()?;
779            match prog.attach(Some("SSL_free"), 0, &libssl_path, None) {
780                Ok(_) => info!("TLS uprobe attached to SSL_free"),
781                Err(e) => warn!("Failed to attach SSL_free uprobe: {}", e),
782            }
783        } else {
784            warn!("libssl.so not found, TLS probes disabled");
785        }
786    }
787
788    info!("All eBPF programs loaded successfully");
789
790    // Take BPF maps for identity and policy
791    let identity_map: AyaHashMap<_, u32, AgentIdentity> =
792        AyaHashMap::try_from(bpf.take_map("AGENT_IDENTITIES").unwrap())?;
793    let identity_map = Arc::new(Mutex::new(identity_map));
794
795    let policy_map: AyaHashMap<_, u32, u8> =
796        AyaHashMap::try_from(bpf.take_map("POLICY_MAP").unwrap())?;
797    let policy_map = Arc::new(Mutex::new(policy_map));
798
799    // TLS connection verdict map  shared between event loop and tracker
800    #[cfg(feature = "tls")]
801    let tls_verdict_map: AyaHashMap<_, TlsConnKey, u8> =
802        AyaHashMap::try_from(bpf.take_map("TLS_CONN_VERDICT").unwrap())?;
803
804    // Kubernetes pod metadata cache (behind feature flag)
805    #[cfg(feature = "k8s")]
806    let k8s_cache = {
807        let cache = Arc::new(RwLock::new(HashMap::<String, k8s::PodMetadata>::new()));
808        let cache_clone = cache.clone();
809        task::spawn(k8s::start_pod_watcher(cache_clone));
810        cache
811    };
812
813    // Event broadcast channel
814    let (tx, _) = broadcast::channel::<ProcessedEvent>(4096);
815
816    // Start CLI output consumer
817    let cli_rx = tx.subscribe();
818    let format = cli.format.clone();
819    task::spawn(cli_output_consumer(cli_rx, format));
820
821    // Start Unix socket server for UI
822    let server_rx = tx.subscribe();
823    task::spawn(server::run_socket_server(server_rx));
824
825    // Start SIEM output consumer if configured
826    if cli.output != "stdout" {
827        if let Some(sink) = siem::OutputSink::parse(&cli.output) {
828            let siem_rx = tx.subscribe();
829            task::spawn(siem::run_siem_consumer(siem_rx, sink));
830        } else {
831            warn!("Unknown output sink format: {}", cli.output);
832        }
833    }
834
835    // Set up RingBuf consumer
836    let ring_buf = RingBuf::try_from(bpf.take_map("EVENTS").unwrap())?;
837    let async_fd = AsyncFd::new(ring_buf)?;
838    let enforce = cli.enforce;
839
840    #[cfg(feature = "k8s")]
841    let k8s_cache_clone = k8s_cache.clone();
842
843    task::spawn(async move {
844        let mut container_cache: HashMap<u32, String> = HashMap::new();
845        let mut pid_stats: HashMap<u32, PidStats> = HashMap::new();
846        let mut async_fd = async_fd;
847        #[cfg(feature = "ml")]
848        let mut ml_classifier = ml::MlClassifier::new();
849        #[cfg(feature = "ml")]
850        let mut ml_last_gc = Instant::now();
851        #[cfg(feature = "ml")]
852        info!("ML behavioral classifier initialized");
853        #[cfg(feature = "prometheus")]
854        let mut llm_pids: HashSet<u32> = HashSet::new();
855        #[cfg(feature = "prometheus")]
856        let mut unique_providers: HashSet<String> = HashSet::new();
857        #[cfg(feature = "tls")]
858        let mut sni_cache = tls::SniCache::new();
859        #[cfg(feature = "tls")]
860        let mut tls_conn_tracker = tls::TlsConnTracker::new();
861        #[cfg(feature = "tls")]
862        let mut tls_verdict_map = tls_verdict_map;
863        #[cfg(feature = "tls")]
864        let mut tls_last_gc = Instant::now();
865
866        loop {
867            let mut guard = match async_fd.readable_mut().await {
868                Ok(g) => g,
869                Err(e) => {
870                    warn!("RingBuf async fd error: {}", e);
871                    break;
872                }
873            };
874            let ring = guard.get_inner_mut();
875            while let Some(item) = ring.next() {
876                let item_len = item.len();
877
878                // Dispatch by event type byte (first byte) and item size
879                // TlsDataEvent: event_type 7 or 8
880                #[cfg(feature = "tls")]
881                if item_len >= std::mem::size_of::<TlsDataEvent>()
882                    && !item.is_empty()
883                    && (item[0] == 7 || item[0] == 8)
884                {
885                    let tls_data =
886                        unsafe { (item.as_ptr() as *const TlsDataEvent).read_unaligned() };
887                    drop(item);
888
889                    let direction_str = if tls_data.direction == 0 {
890                        "write"
891                    } else {
892                        "read"
893                    };
894                    let key = TlsConnKey {
895                        pid: tls_data.pid,
896                        _pad: 0,
897                        ssl_ptr: tls_data.ssl_ptr,
898                    };
899
900                    // Track chunk count
901                    let chunk_num = tls_conn_tracker.record_chunk(tls_data.pid, tls_data.ssl_ptr);
902
903                    // Get SNI hint for this PID
904                    let sni_hint = sni_cache.get(tls_data.pid);
905
906                    if tls_conn_tracker.is_decided(tls_data.pid, tls_data.ssl_ptr) {
907                        // Already decided interesting  classify and forward
908                        let classification = tls::classify_payload(
909                            tls_data.payload_bytes(),
910                            tls_data.direction,
911                            sni_hint,
912                        );
913                        debug!(
914                            "TLS data {}: PID {} ({}) {} bytes",
915                            direction_str,
916                            tls_data.pid,
917                            tls_data.process_name(),
918                            tls_data.payload_len,
919                        );
920
921                        let processed = events::from_tls_data_event(&tls_data, &classification);
922                        let _ = tx.send(processed);
923                    } else {
924                        // Still undecided  classify this chunk
925                        let classification = tls::classify_payload(
926                            tls_data.payload_bytes(),
927                            tls_data.direction,
928                            sni_hint,
929                        );
930
931                        if classification.is_interesting {
932                            // Found LLM/MCP traffic!
933                            tls_conn_tracker.set_verdict(tls_data.pid, tls_data.ssl_ptr, true);
934                            let _ = tls_verdict_map.insert(key, 1u8, 0); // INTERESTING
935
936                            #[cfg(feature = "prometheus")]
937                            {
938                                metrics::record_tls_verdict("interesting");
939                                metrics::record_classifier_confidence(classification.confidence);
940                            }
941
942                            info!(
943                                "TLS: PID {} ({}) -> {} {} [chunk #{}]",
944                                tls_data.pid,
945                                tls_data.process_name(),
946                                classification.content_class_str().unwrap_or("unknown"),
947                                classification.provider().unwrap_or(""),
948                                chunk_num,
949                            );
950
951                            let processed = events::from_tls_data_event(&tls_data, &classification);
952                            let _ = tx.send(processed);
953                        } else if tls_conn_tracker
954                            .should_mark_boring(tls_data.pid, tls_data.ssl_ptr)
955                        {
956                            // Hit the limit  mark as boring
957                            tls_conn_tracker.set_verdict(tls_data.pid, tls_data.ssl_ptr, false);
958                            let _ = tls_verdict_map.insert(key, 2u8, 0); // BORING
959
960                            #[cfg(feature = "prometheus")]
961                            metrics::record_tls_verdict("boring");
962
963                            debug!(
964                                "TLS: PID {} ({}) -> boring after {} chunks",
965                                tls_data.pid,
966                                tls_data.process_name(),
967                                chunk_num,
968                            );
969                        } else {
970                            debug!(
971                                "TLS {} (undecided #{}) PID {} ({}) {} bytes",
972                                direction_str,
973                                chunk_num,
974                                tls_data.pid,
975                                tls_data.process_name(),
976                                tls_data.payload_len,
977                            );
978                        }
979                    }
980
981                    continue;
982                }
983
984                // TlsHandshakeEvent: event_type 6
985                #[cfg(feature = "tls")]
986                if item_len >= std::mem::size_of::<TlsHandshakeEvent>()
987                    && !item.is_empty()
988                    && item[0] == 6
989                {
990                    let tls_event =
991                        unsafe { (item.as_ptr() as *const TlsHandshakeEvent).read_unaligned() };
992                    let sni = tls_event.sni_str().to_string();
993                    if !sni.is_empty() {
994                        info!(
995                            "TLS SNI: PID {} ({}) -> {}",
996                            tls_event.pid,
997                            tls_event.process_name(),
998                            sni,
999                        );
1000                        sni_cache.insert(tls_event.pid, sni);
1001                    }
1002                    drop(item);
1003                    continue;
1004                }
1005
1006                if item_len >= std::mem::size_of::<NetworkEvent>() {
1007                    let event = unsafe { (item.as_ptr() as *const NetworkEvent).read_unaligned() };
1008                    drop(item); // consume the ring buf entry before async work
1009                    handle_event(
1010                        event,
1011                        &tx,
1012                        &identity_map,
1013                        &policy_map,
1014                        &provider_map,
1015                        enforce,
1016                        &mut container_cache,
1017                        &mut pid_stats,
1018                        #[cfg(feature = "k8s")]
1019                        &k8s_cache_clone,
1020                        #[cfg(feature = "ml")]
1021                        &mut ml_classifier,
1022                        #[cfg(feature = "tls")]
1023                        &sni_cache,
1024                        #[cfg(feature = "prometheus")]
1025                        &mut llm_pids,
1026                        #[cfg(feature = "prometheus")]
1027                        &mut unique_providers,
1028                    )
1029                    .await;
1030                }
1031            }
1032            guard.clear_ready();
1033
1034            // Periodic ML idle PID garbage collection
1035            #[cfg(feature = "ml")]
1036            if ml_last_gc.elapsed() >= Duration::from_secs(60) {
1037                ml_classifier.gc_idle_pids(Duration::from_secs(300));
1038                ml_last_gc = Instant::now();
1039            }
1040
1041            // Periodic SNI cache + TLS connection tracker garbage collection
1042            #[cfg(feature = "tls")]
1043            if tls_last_gc.elapsed() >= Duration::from_secs(60) {
1044                sni_cache.gc();
1045                tls_conn_tracker.gc();
1046                #[cfg(feature = "prometheus")]
1047                metrics::set_tls_connections_tracked(tls_conn_tracker.len());
1048                tls_last_gc = Instant::now();
1049            }
1050        }
1051    });
1052
1053    info!("Monitoring LLM/AI communications... Press Ctrl-C to exit");
1054
1055    signal::ctrl_c().await?;
1056    info!("Exiting...");
1057
1058    // Clean up socket file
1059    let _ = std::fs::remove_file("/tmp/busted.sock");
1060
1061    Ok(())
1062}
