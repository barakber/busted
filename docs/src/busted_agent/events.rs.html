4/// Create a BustedEvent from a raw network event.
5pub fn from_network_event(
6    event: &NetworkEvent,
7    provider: Option<&str>,
8    policy: Option<&str>,
9) -> BustedEvent {
10    let kind = match event.event_type {
11        1 => NetworkEventKind::Connect,
12        2 => NetworkEventKind::DataSent,
13        3 => NetworkEventKind::DataReceived,
14        4 => NetworkEventKind::Close,
15        5 => NetworkEventKind::DnsQuery,
16        _ => NetworkEventKind::Connect,
17    };
18
19    BustedEvent {
20        timestamp: format_timestamp(event.timestamp_ns),
21        process: ProcessInfo {
22            pid: event.pid,
23            uid: event.uid,
24            name: event.process_name().to_string(),
25            container_id: event.container_id_str().to_string(),
26            cgroup_id: event.cgroup_id,
27            pod_name: None,
28            pod_namespace: None,
29            service_account: None,
30        },
31        session_id: format!("{}:net", event.pid),
32        identity: None,
33        policy: policy.map(|s| s.to_string()),
34        action: AgenticAction::Network {
35            kind,
36            src_ip: event.source_ip().to_string(),
37            src_port: event.sport,
38            dst_ip: event.dest_ip().to_string(),
39            dst_port: event.dport,
40            bytes: event.bytes,
41            sni: None,
42            provider: provider.map(|s| s.to_string()),
43        },
44    }
45}

75pub fn format_timestamp(ns: u64) -> String {
76    use std::time::{Duration, SystemTime, UNIX_EPOCH};
77
78    // bpf_ktime_get_ns returns time since boot, not epoch.
79    // Convert by computing the boot offset from current wall clock.
80    let boot_offset = SystemTime::now()
81        .duration_since(UNIX_EPOCH)
82        .unwrap_or_default()
83        .saturating_sub(Duration::from_nanos(ns));
84
85    let wall_ns = boot_offset.as_nanos() as u64 + ns;
86    let secs = wall_ns / 1_000_000_000;
87    let subsec = (wall_ns % 1_000_000_000) as u32;
88
89    let total_secs = secs;
90    let hours = (total_secs / 3600) % 24;
91    let minutes = (total_secs / 60) % 60;
92    let seconds = total_secs % 60;
93    let millis = subsec / 1_000_000;
94
95    format!("{:02}:{:02}:{:02}.{:03}", hours, minutes, seconds, millis)
96}

98pub fn format_timestamp_now() -> String {
99    use std::time::{SystemTime, UNIX_EPOCH};
100
101    let now = SystemTime::now()
102        .duration_since(UNIX_EPOCH)
103        .unwrap_or_default();
104    let secs = now.as_secs();
105    let millis = now.subsec_millis();
106    let hours = (secs / 3600) % 24;
107    let minutes = (secs / 60) % 60;
108    let seconds = secs % 60;
109
110    format!("{:02}:{:02}:{:02}.{:03}", hours, minutes, seconds, millis)
